22/// GeneratorBuilder can be used to configure Responder Interceptor
23#[derive(Default)]
24pub struct ResponderBuilder {
25    log2_size: Option<u8>,
26}
27
28impl ResponderBuilder {
29    /// with_log2_size sets the size of the interceptor.
30    /// Size must be one of: 1, 2, 4, 8, 16, 32, 64, 128, 256, 512, 1024, 2048, 4096, 8192, 16384, 32768
31    pub fn with_log2_size(mut self, log2_size: u8) -> ResponderBuilder {
32        self.log2_size = Some(log2_size);
33        self
34    }
35}
36
37impl InterceptorBuilder for ResponderBuilder {
38    fn build(&self, _id: &str) -> Result<Arc<dyn Interceptor + Send + Sync>> {
39        Ok(Arc::new(Responder {
40            internal: Arc::new(ResponderInternal {
41                log2_size: self.log2_size.unwrap_or(13), // 8192 = 1 << 13
42                streams: Arc::new(Mutex::new(HashMap::new())),
43            }),
44        }))
45    }
46}

48pub struct ResponderInternal {
49    log2_size: u8,
50    streams: Arc<Mutex<HashMap<u32, Arc<ResponderStream>>>>,
51}
52
53impl ResponderInternal {
54    async fn resend_packets(
55        streams: Arc<Mutex<HashMap<u32, Arc<ResponderStream>>>>,
56        nack: TransportLayerNack,
57    ) {
58        let stream = {
59            let m = streams.lock().await;
60            if let Some(stream) = m.get(&nack.media_ssrc) {
61                stream.clone()
62            } else {
63                return;
64            }
65        };
66
67        for n in &nack.nacks {
68            // can't use n.range() since this callback is async fn,
69            // instead, use NackPair into_iter()
70            let stream2 = Arc::clone(&stream);
71            let f = Box::new(
72                move |seq: u16| -> Pin<Box<dyn Future<Output = bool> + Send + 'static>> {
73                    let stream3 = Arc::clone(&stream2);
74                    Box::pin(async move {
75                        if let Some(p) = stream3.get(seq).await {
76                            let a = Attributes::new();
77                            if let Err(err) = stream3.next_rtp_writer.write(&p, &a).await {
78                                log::warn!("failed resending nacked packet: {}", err);
79                            }
80                        }
81                        true
82                    })
83                },
84            );
85            for packet_id in n.into_iter() {
86                if !f(packet_id).await {
87                    return;
88                }
89            }
90        }
91    }
92}

94pub struct ResponderRtcpReader {
95    parent_rtcp_reader: Arc<dyn RTCPReader + Send + Sync>,
96    internal: Arc<ResponderInternal>,
97}
98
99#[async_trait]
100impl RTCPReader for ResponderRtcpReader {
101    async fn read(
102        &self,
103        buf: &mut [u8],
104        a: &Attributes,
105    ) -> Result<(Vec<Box<dyn rtcp::packet::Packet + Send + Sync>>, Attributes)> {
106        let (pkts, attr) = { self.parent_rtcp_reader.read(buf, a).await? };
107        for p in &pkts {
108            if let Some(nack) = p.as_any().downcast_ref::<TransportLayerNack>() {
109                let nack = nack.clone();
110                let streams = Arc::clone(&self.internal.streams);
111                tokio::spawn(async move {
112                    ResponderInternal::resend_packets(streams, nack).await;
113                });
114            }
115        }
116
117        Ok((pkts, attr))
118    }
119}

121/// Responder responds to nack feedback messages
122pub struct Responder {
123    internal: Arc<ResponderInternal>,
124}
125
126impl Responder {
127    /// builder returns a new ResponderBuilder.
128    pub fn builder() -> ResponderBuilder {
129        ResponderBuilder::default()
130    }
131}
132
133#[async_trait]
134impl Interceptor for Responder {
135    /// bind_rtcp_reader lets you modify any incoming RTCP packets. It is called once per sender/receiver, however this might
136    /// change in the future. The returned method will be called once per packet batch.
137    async fn bind_rtcp_reader(
138        &self,
139        reader: Arc<dyn RTCPReader + Send + Sync>,
140    ) -> Arc<dyn RTCPReader + Send + Sync> {
141        Arc::new(ResponderRtcpReader {
142            internal: Arc::clone(&self.internal),
143            parent_rtcp_reader: reader,
144        }) as Arc<dyn RTCPReader + Send + Sync>
145    }
146
147    /// bind_rtcp_writer lets you modify any outgoing RTCP packets. It is called once per PeerConnection. The returned method
148    /// will be called once per packet batch.
149    async fn bind_rtcp_writer(
150        &self,
151        writer: Arc<dyn RTCPWriter + Send + Sync>,
152    ) -> Arc<dyn RTCPWriter + Send + Sync> {
153        writer
154    }
155
156    /// bind_local_stream lets you modify any outgoing RTP packets. It is called once for per LocalStream. The returned method
157    /// will be called once per rtp packet.
158    async fn bind_local_stream(
159        &self,
160        info: &StreamInfo,
161        writer: Arc<dyn RTPWriter + Send + Sync>,
162    ) -> Arc<dyn RTPWriter + Send + Sync> {
163        if !stream_support_nack(info) {
164            return writer;
165        }
166
167        let stream = Arc::new(ResponderStream::new(self.internal.log2_size, writer));
168        {
169            let mut streams = self.internal.streams.lock().await;
170            streams.insert(info.ssrc, Arc::clone(&stream));
171        }
172
173        stream
174    }
175
176    /// unbind_local_stream is called when the Stream is removed. It can be used to clean up any data related to that track.
177    async fn unbind_local_stream(&self, info: &StreamInfo) {
178        let mut streams = self.internal.streams.lock().await;
179        streams.remove(&info.ssrc);
180    }
181
182    /// bind_remote_stream lets you modify any incoming RTP packets. It is called once for per RemoteStream. The returned method
183    /// will be called once per rtp packet.
184    async fn bind_remote_stream(
185        &self,
186        _info: &StreamInfo,
187        reader: Arc<dyn RTPReader + Send + Sync>,
188    ) -> Arc<dyn RTPReader + Send + Sync> {
189        reader
190    }
191
192    /// unbind_remote_stream is called when the Stream is removed. It can be used to clean up any data related to that track.
193    async fn unbind_remote_stream(&self, _info: &StreamInfo) {}
194
195    /// close closes the Interceptor, cleaning up any data if necessary.
196    async fn close(&self) -> Result<()> {
197        Ok(())
198    }
199}
