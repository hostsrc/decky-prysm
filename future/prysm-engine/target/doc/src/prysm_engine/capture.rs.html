1//! FFmpeg capture supervisor
2//!
3//! Spawns FFmpeg with kmsgrab → VAAPI H.264 → RTP output to localhost.
4//! Monitors the process and auto-restarts when Gamescope changes the
5//! DRM framebuffer format (which kills kmsgrab).
6//!
7//! Future v2: Replace FFmpeg with direct DRM/KMS + VAAPI via libdrm/libva FFI.
8
9use std::process::Stdio;
10use std::sync::Arc;
11use tokio::io::{AsyncBufReadExt, BufReader};
12use tokio::net::UdpSocket;
13use tokio::process::{Child, Command};
14use tokio::sync::broadcast;
15use tokio::task::JoinHandle;
16
17use crate::stats::PipelineStats;

19const RTP_RECV_PORT: u16 = 5004;
20const MAX_RESTARTS: u32 = 500;
21
22#[derive(Debug, Clone)]
23pub struct QualityPreset {
24    pub width: u32,
25    pub height: u32,
26    pub fps: u32,
27    pub bitrate_kbps: u32,
28}
29
30impl QualityPreset {
31    pub fn from_name(name: &str) -> Self {
32        match name {
33            "480p30" => Self { width: 854, height: 480, fps: 30, bitrate_kbps: 2500 },
34            "720p30" => Self { width: 1280, height: 720, fps: 30, bitrate_kbps: 5000 },
35            "720p60" => Self { width: 1280, height: 720, fps: 60, bitrate_kbps: 8000 },
36            "1080p30" => Self { width: 1920, height: 1080, fps: 30, bitrate_kbps: 8000 },
37            "1080p60" => Self { width: 1920, height: 1080, fps: 60, bitrate_kbps: 12000 },
38            _ => Self { width: 1280, height: 720, fps: 30, bitrate_kbps: 5000 },
39        }
40    }
41}

43/// Build the FFmpeg command that outputs RTP to localhost UDP.
44fn build_ffmpeg_cmd(preset: &QualityPreset, audio: bool) -> Vec<String> {
45    let mut args: Vec<String> = vec![
46        "ffmpeg".into(),
47        "-y".into(),
48        "-fflags".into(), "nobuffer".into(),
49        "-flags".into(), "low_delay".into(),
50        // Video input: KMS/DRM
51        "-device".into(), "/dev/dri/card0".into(),
52        "-framerate".into(), preset.fps.to_string(),
53        "-f".into(), "kmsgrab".into(),
54        "-i".into(), "-".into(),
55    ];
56
57    // Video encode: VAAPI H.264, minimal latency
58    let vf = format!(
59        "hwmap=derive_device=vaapi,scale_vaapi=w={}:h={}:format=nv12",
60        preset.width, preset.height
61    );
62    args.extend([
63        "-vaapi_device".into(), "/dev/dri/renderD128".into(),
64        "-vf".into(), vf,
65        "-c:v".into(), "h264_vaapi".into(),
66        "-b:v".into(), format!("{}k", preset.bitrate_kbps),
67        "-maxrate".into(), format!("{}k", preset.bitrate_kbps),
68        "-bufsize".into(), format!("{}k", preset.bitrate_kbps / 2),
69        "-g".into(), preset.fps.to_string(),
70        "-bf".into(), "0".into(),
71    ]);
72
73    // Output: raw H.264 RTP to localhost (prysm-engine reads and relays to WebRTC)
74    // Audio will be added in v2 as a separate WebRTC track
75    args.extend([
76        "-f".into(), "rtp".into(),
77        "-flush_packets".into(), "1".into(),
78        "-sdp_file".into(), "/tmp/prysm/stream.sdp".into(),
79        format!("rtp://127.0.0.1:{RTP_RECV_PORT}"),
80    ]);
81
82    args
83}

85/// Spawn FFmpeg process.
86fn spawn_ffmpeg(preset: &QualityPreset, audio: bool) -> anyhow::Result<Child> {
87    let args = build_ffmpeg_cmd(preset, audio);
88    tracing::info!("FFmpeg cmd: {}", args.join(" "));
89
90    let child = Command::new(&args[0])
91        .args(&args[1..])
92        .env("XDG_RUNTIME_DIR", "/run/user/1000")
93        .env("PULSE_SERVER", "/run/user/1000/pulse/native")
94        .env("HOME", "/home/deck")
95        .stdout(Stdio::null())
96        .stderr(Stdio::piped())
97        .kill_on_drop(true)
98        .spawn()?;
99
100    tracing::info!("FFmpeg started (PID {})", child.id().unwrap_or(0));
101    Ok(child)
102}

104/// Start the capture supervisor + RTP receiver.
105///
106/// Returns a broadcast receiver for RTP packets and a join handle for the supervisor task.
107pub async fn start_capture_supervisor(
108    preset: QualityPreset,
109    audio: bool,
110    stats: Arc<PipelineStats>,
111) -> anyhow::Result<(broadcast::Receiver<Vec<u8>>, JoinHandle<()>)> {
112    // Broadcast channel: FFmpeg → RTP packets → all WebRTC peers
113    // Large buffer to handle keyframe bursts (IDR can be 50+ packets)
114    let (tx, rx) = broadcast::channel::<Vec<u8>>(2048);
115
116    // UDP socket to receive RTP from FFmpeg
117    let sock = UdpSocket::bind(format!("127.0.0.1:{RTP_RECV_PORT}")).await?;
118    tracing::info!("RTP receiver listening on 127.0.0.1:{RTP_RECV_PORT}");
119
120    // Spawn the RTP reader task
121    let tx_rtp = tx.clone();
122    let rtp_stats = stats.clone();
123    tokio::spawn(async move {
124        let mut buf = vec![0u8; 65536];
125        loop {
126            match sock.recv(&mut buf).await {
127                Ok(n) if n > 0 => {
128                    rtp_stats.rtp_packets_recv.inc();
129                    rtp_stats.rtp_bytes_recv.add(n as u64);
130                    let _ = tx_rtp.send(buf[..n].to_vec());
131                }
132                Ok(_) => {}
133                Err(e) => {
134                    rtp_stats.rtp_recv_errors.inc();
135                    tracing::warn!("RTP recv error: {e}");
136                    tokio::time::sleep(std::time::Duration::from_millis(100)).await;
137                }
138            }
139        }
140    });
141
142    // Spawn the FFmpeg supervisor task
143    let sup_stats = stats.clone();
144    let handle = tokio::spawn(async move {
145        let mut restart_count: u32 = 0;
146
147        loop {
148            let mut child = match spawn_ffmpeg(&preset, audio) {
149                Ok(c) => c,
150                Err(e) => {
151                    tracing::error!("Failed to spawn FFmpeg: {e}");
152                    tokio::time::sleep(std::time::Duration::from_secs(2)).await;
153                    restart_count += 1;
154                    if restart_count > MAX_RESTARTS {
155                        tracing::error!("Too many FFmpeg restarts, giving up");
156                        return;
157                    }
158                    continue;
159                }
160            };
161
162            // Monitor stderr for the kmsgrab format change error
163            if let Some(stderr) = child.stderr.take() {
164                let reader = BufReader::new(stderr);
165                let mut lines = reader.lines();
166
167                loop {
168                    tokio::select! {
169                        // Check if process exited
170                        status = child.wait() => {
171                            match status {
172                                Ok(s) => tracing::warn!("FFmpeg exited: {s}"),
173                                Err(e) => tracing::warn!("FFmpeg wait error: {e}"),
174                            }
175                            break;
176                        }
177                        // Watch stderr for format change (proactive restart)
178                        line = lines.next_line() => {
179                            match line {
180                                Ok(Some(l)) if l.contains("framebuffer format changed") => {
181                                    tracing::warn!("Gamescope format change detected — killing FFmpeg");
182                                    sup_stats.ffmpeg_format_changes.inc();
183                                    let _ = child.kill().await;
184                                    break;
185                                }
186                                Ok(Some(_)) => {} // normal stderr output
187                                Ok(None) => break, // EOF
188                                Err(_) => break,
189                            }
190                        }
191                    }
192                }
193            } else {
194                // No stderr handle, just wait for exit
195                let _ = child.wait().await;
196            }
197
198            restart_count += 1;
199            sup_stats.ffmpeg_restarts.inc();
200            tracing::info!("Restarting FFmpeg (attempt {restart_count})");
201            // Brief pause before restart
202            tokio::time::sleep(std::time::Duration::from_millis(500)).await;
203        }
204    });
205
206    Ok((rx, handle))
207}
