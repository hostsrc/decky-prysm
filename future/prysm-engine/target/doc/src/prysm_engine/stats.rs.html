1//! Pipeline performance profiler
2//!
3//! Tracks metrics at every stage of the streaming pipeline:
4//!   FFmpeg capture → RTP recv → WebRTC relay → Browser
5//!
6//! Exposes stats via /stats JSON endpoint and periodic log output.
7
8use std::sync::atomic::{AtomicU64, Ordering};
9use std::sync::Arc;
10use std::time::Instant;
11use serde::Serialize;
12use tokio::sync::RwLock;
13
14/// Atomic counter helper
15pub struct Counter(AtomicU64);
16impl Counter {
17    pub const fn new() -> Self { Self(AtomicU64::new(0)) }
18    pub fn inc(&self) { self.0.fetch_add(1, Ordering::Relaxed); }
19    pub fn add(&self, n: u64) { self.0.fetch_add(n, Ordering::Relaxed); }
20    pub fn get(&self) -> u64 { self.0.load(Ordering::Relaxed) }
21}
22
23/// All pipeline stats — updated atomically from multiple tasks
24pub struct PipelineStats {
25    start_time: Instant,
26
27    // FFmpeg supervisor
28    pub ffmpeg_restarts: Counter,
29    pub ffmpeg_format_changes: Counter,
30
31    // RTP receiver (FFmpeg → engine)
32    pub rtp_packets_recv: Counter,
33    pub rtp_bytes_recv: Counter,
34    pub rtp_recv_errors: Counter,
35
36    // Broadcast channel
37    pub broadcast_lagged: Counter,
38
39    // WebRTC relay (engine → browser)
40    pub webrtc_packets_sent: Counter,
41    pub webrtc_bytes_sent: Counter,
42    pub webrtc_write_errors: Counter,
43    pub webrtc_peers_total: Counter,
44
45    // WHEP signaling
46    pub whep_offers: Counter,
47    pub whep_errors: Counter,
48
49    // Snapshot for rate calculations
50    snapshot: RwLock<StatsSnapshot>,
51}
52
53struct StatsSnapshot {
54    timestamp: Instant,
55    rtp_packets: u64,
56    rtp_bytes: u64,
57    webrtc_packets: u64,
58    webrtc_bytes: u64,
59}
60
61/// JSON-serializable stats response
62#[derive(Serialize, Clone)]
63pub struct StatsReport {
64    pub uptime_secs: f64,
65
66    // FFmpeg
67    pub ffmpeg_restarts: u64,
68    pub ffmpeg_format_changes: u64,
69
70    // RTP ingest (packets per second, megabits per second)
71    pub rtp_pps: f64,
72    pub rtp_mbps: f64,
73    pub rtp_total_packets: u64,
74    pub rtp_total_bytes: u64,
75    pub rtp_errors: u64,
76
77    // Broadcast
78    pub broadcast_lagged: u64,
79
80    // WebRTC egress
81    pub webrtc_pps: f64,
82    pub webrtc_mbps: f64,
83    pub webrtc_total_packets: u64,
84    pub webrtc_errors: u64,
85    pub webrtc_peers_total: u64,
86
87    // WHEP
88    pub whep_offers: u64,
89    pub whep_errors: u64,
90}
91
92impl PipelineStats {
93    pub fn new() -> Arc<Self> {
94        let now = Instant::now();
95        Arc::new(Self {
96            start_time: now,
97            ffmpeg_restarts: Counter::new(),
98            ffmpeg_format_changes: Counter::new(),
99            rtp_packets_recv: Counter::new(),
100            rtp_bytes_recv: Counter::new(),
101            rtp_recv_errors: Counter::new(),
102            broadcast_lagged: Counter::new(),
103            webrtc_packets_sent: Counter::new(),
104            webrtc_bytes_sent: Counter::new(),
105            webrtc_write_errors: Counter::new(),
106            webrtc_peers_total: Counter::new(),
107            whep_offers: Counter::new(),
108            whep_errors: Counter::new(),
109            snapshot: RwLock::new(StatsSnapshot {
110                timestamp: now,
111                rtp_packets: 0,
112                rtp_bytes: 0,
113                webrtc_packets: 0,
114                webrtc_bytes: 0,
115            }),
116        })
117    }
118
119    /// Generate a stats report with rate calculations
120    pub async fn report(&self) -> StatsReport {
121        let now = Instant::now();
122        let uptime = now.duration_since(self.start_time).as_secs_f64();
123
124        let rtp_packets = self.rtp_packets_recv.get();
125        let rtp_bytes = self.rtp_bytes_recv.get();
126        let webrtc_packets = self.webrtc_packets_sent.get();
127
128        // Calculate rates from snapshot delta
129        let mut snap = self.snapshot.write().await;
130        let dt = now.duration_since(snap.timestamp).as_secs_f64().max(0.001);
131        let rtp_pps = (rtp_packets - snap.rtp_packets) as f64 / dt;
132        let rtp_mbps = ((rtp_bytes - snap.rtp_bytes) as f64 * 8.0) / (dt * 1_000_000.0);
133        let webrtc_pps = (webrtc_packets - snap.webrtc_packets) as f64 / dt;
134        let webrtc_bytes = self.webrtc_bytes_sent.get();
135        let webrtc_mbps = ((webrtc_bytes - snap.webrtc_bytes) as f64 * 8.0) / (dt * 1_000_000.0);
136
137        // Update snapshot
138        *snap = StatsSnapshot {
139            timestamp: now,
140            rtp_packets,
141            rtp_bytes,
142            webrtc_packets,
143            webrtc_bytes,
144        };
145
146        StatsReport {
147            uptime_secs: uptime,
148            ffmpeg_restarts: self.ffmpeg_restarts.get(),
149            ffmpeg_format_changes: self.ffmpeg_format_changes.get(),
150            rtp_pps,
151            rtp_mbps,
152            rtp_total_packets: rtp_packets,
153            rtp_total_bytes: rtp_bytes,
154            rtp_errors: self.rtp_recv_errors.get(),
155            broadcast_lagged: self.broadcast_lagged.get(),
156            webrtc_pps,
157            webrtc_mbps,
158            webrtc_total_packets: webrtc_packets,
159            webrtc_errors: self.webrtc_write_errors.get(),
160            webrtc_peers_total: self.webrtc_peers_total.get(),
161            whep_offers: self.whep_offers.get(),
162            whep_errors: self.whep_errors.get(),
163        }
164    }
165}

167/// Spawn a background task that logs stats every N seconds
168pub fn spawn_stats_logger(stats: Arc<PipelineStats>, interval_secs: u64) {
169    tokio::spawn(async move {
170        let mut interval = tokio::time::interval(std::time::Duration::from_secs(interval_secs));
171        loop {
172            interval.tick().await;
173            let r = stats.report().await;
174            tracing::info!(
175                "STATS | up:{:.0}s | rtp:{:.0}pps {:.1}Mbps ({} pkts, {} err) | webrtc:{:.0}pps {:.1}Mbps ({} err) | peers:{} | ffmpeg_restarts:{} | lagged:{} | whep:{}/{}",
176                r.uptime_secs,
177                r.rtp_pps, r.rtp_mbps, r.rtp_total_packets, r.rtp_errors,
178                r.webrtc_pps, r.webrtc_mbps, r.webrtc_errors,
179                r.webrtc_peers_total,
180                r.ffmpeg_restarts,
181                r.broadcast_lagged,
182                r.whep_offers, r.whep_errors,
183            );
184        }
185    });
