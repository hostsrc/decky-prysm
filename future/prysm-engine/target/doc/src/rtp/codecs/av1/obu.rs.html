4use bytes::Bytes;

6use crate::codecs::av1::leb128::read_leb128;

10pub const OBU_HAS_EXTENSION_BIT: u8 = 0b0000_0100;
11pub const OBU_HAS_SIZE_BIT: u8 = 0b0000_0010;
12pub const OBU_TYPE_MASK: u8 = 0b0111_1000;
13
14pub const OBU_TYPE_SEQUENCE_HEADER: u8 = 1;
15pub const OBU_TYPE_TEMPORAL_DELIMITER: u8 = 2;
16pub const OBU_TYPE_FRAME_HEADER: u8 = 3;
17pub const OBU_TYPE_TILE_GROUP: u8 = 4;
18pub const OBU_TYPE_METADATA: u8 = 5;
19pub const OBU_TYPE_FRAME: u8 = 6;
20pub const OBU_TYPE_TILE_LIST: u8 = 8;
21pub const OBU_TYPE_PADDING: u8 = 15;
22
23#[derive(Debug, Clone)]
24pub struct Obu {
25    pub header: u8,
26    pub extension_header: u8,
27    pub payload: Bytes,
28    /// size of the header and payload combined.
29    pub size: usize,
30}
31
32impl Obu {
33    pub fn header_size(&self) -> usize {
34        if obu_has_extension(self.header) {
35            2
36        } else {
37            1
38        }
39    }
40}

42/// Parses the payload into series of OBUs.
43/// Reference: https://aomediacodec.github.io/av1-spec/#obu-syntax
44pub fn parse_obus(payload: &Bytes) -> Result<Vec<Obu>> {
45    let mut obus = vec![];
46    let mut payload_data_remaining = payload.len() as isize;
47    let mut payload_data_index: usize = 0;
48
49    while payload_data_remaining > 0 {
50        // Read OBU header.
51        let header = payload[payload_data_index];
52        let has_extension = obu_has_extension(header);
53        let has_size = obu_has_size(header);
54        let obu_type = obu_type(header);
55
56        // Read OBU extension header.
57        let extension_header = if has_extension {
58            if payload_data_remaining < 2 {
59                return Err(ErrPayloadTooSmallForObuExtensionHeader);
60            }
61            payload[payload_data_index + 1]
62        } else {
63            0
64        };
65        let obu_header_size = if has_extension { 2 } else { 1 };
66        let payload_without_header = payload.slice(payload_data_index + obu_header_size..);
67
68        // Read OBU payload.
69        let obu_payload = if !has_size {
70            payload_without_header
71        } else {
72            if payload_without_header.is_empty() {
73                return Err(ErrPayloadTooSmallForObuPayloadSize);
74            }
75            let (obu_payload_size, leb128_size) = read_leb128(&payload_without_header);
76            payload_data_remaining -= leb128_size as isize;
77            payload_data_index += leb128_size;
78            payload_without_header.slice(leb128_size..leb128_size + obu_payload_size as usize)
79        };
80
81        let obu_size = obu_header_size + obu_payload.len();
82        if !should_ignore_obu_type(obu_type) {
83            obus.push(Obu {
84                header,
85                extension_header,
86                payload: obu_payload,
87                size: obu_size,
88            });
89        }
90
91        payload_data_remaining -= obu_size as isize;
92        payload_data_index += obu_size;
93    }
94
95    Ok(obus)
96}
97
98pub fn obu_has_extension(header: u8) -> bool {
99    header & OBU_HAS_EXTENSION_BIT != 0
100}
101
102pub fn obu_has_size(header: u8) -> bool {
103    header & OBU_HAS_SIZE_BIT != 0
104}
105
106pub fn obu_type(header: u8) -> u8 {
107    (header & OBU_TYPE_MASK) >> 3
108}
109
110fn should_ignore_obu_type(obu_type: u8) -> bool {
111    obu_type == OBU_TYPE_TEMPORAL_DELIMITER
112        || obu_type == OBU_TYPE_TILE_LIST
113        || obu_type == OBU_TYPE_PADDING
114}
