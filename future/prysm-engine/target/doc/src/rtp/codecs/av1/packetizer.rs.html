9/// When there are 3 or less OBU (fragments) in a packet, size of the last one
10/// can be omitted.
11pub const MAX_NUM_OBUS_TO_OMIT_SIZE: usize = 3;
12pub const AGGREGATION_HEADER_SIZE: usize = 1;
13
14pub struct PacketMetadata {
15    pub first_obu_index: usize,
16    pub num_obu_elements: usize,
17    pub first_obu_offset: usize,
18    pub last_obu_size: usize,
19    /// Total size consumed by the packet.
20    pub packet_size: usize,
21}
22
23impl PacketMetadata {
24    fn new(first_obu_index: usize) -> Self {
25        Self {
26            first_obu_index,
27            num_obu_elements: 0,
28            first_obu_offset: 0,
29            last_obu_size: 0,
30            packet_size: 0,
31        }
32    }
33}

35/// Returns the scheme for how to aggregate or split the OBUs across RTP packets.
36/// Reference: https://aomediacodec.github.io/av1-rtp-spec/#45-payload-structure
37///            https://aomediacodec.github.io/av1-rtp-spec/#5-packetization-rules
38pub fn packetize(obus: &[Obu], mtu: usize) -> Vec<PacketMetadata> {
39    if obus.is_empty() {
40        return vec![];
41    }
42    // Ignore certain edge cases where packets should be very small. They are
43    // impractical but adds complexity to handle.
44    if mtu < 3 {
45        return vec![];
46    }
47
48    let mut packets = vec![];
49
50    // Aggregation header will be present in all packets.
51    let max_payload_size = mtu - AGGREGATION_HEADER_SIZE;
52
53    // Assemble packets. Push to current packet as much as it can hold before
54    // considering next one. That would normally cause uneven distribution across
55    // packets, specifically last one would be generally smaller.
56    packets.push(PacketMetadata::new(0));
57    let mut packet_remaining_bytes = max_payload_size;
58
59    for obu_index in 0..obus.len() {
60        let is_last_obu = obu_index == obus.len() - 1;
61        let obu = &obus[obu_index];
62
63        // Putting |obu| into the last packet would make last obu element stored in
64        // that packet not last. All not last OBU elements must be prepend with the
65        // element length. AdditionalBytesForPreviousObuElement calculates how many
66        // bytes are needed to store that length.
67        let mut packet = packets.pop().unwrap();
68        let mut previous_obu_extra_size = additional_bytes_for_previous_obu_element(&packet);
69        let min_required_size = if packet.num_obu_elements >= MAX_NUM_OBUS_TO_OMIT_SIZE {
70            2
71        } else {
72            1
73        };
74        if packet_remaining_bytes < previous_obu_extra_size + min_required_size {
75            // Start a new packet.
76            packets.push(packet);
77            packet = PacketMetadata::new(obu_index);
78            packet_remaining_bytes = max_payload_size;
79            previous_obu_extra_size = 0;
80        }
81        packet.packet_size += previous_obu_extra_size;
82        packet_remaining_bytes -= previous_obu_extra_size;
83        packet.num_obu_elements += 1;
84        let must_write_obu_element_size = packet.num_obu_elements > MAX_NUM_OBUS_TO_OMIT_SIZE;
85
86        // Can fit all of the obu into the packet?
87        let mut required_bytes = obu.size;
88        if must_write_obu_element_size {
89            required_bytes += leb128_size(obu.size as u32);
90        }
91        if required_bytes < packet_remaining_bytes {
92            // Insert the obu into the packet unfragmented.
93            packet.last_obu_size = obu.size;
94            packet.packet_size += required_bytes;
95            packet_remaining_bytes -= required_bytes;
96            packets.push(packet);
97            continue;
98        }
99
100        // Fragment the obu.
101        let max_first_fragment_size = if must_write_obu_element_size {
102            max_fragment_size(packet_remaining_bytes)
103        } else {
104            packet_remaining_bytes
105        };
106        // Because available_bytes might be different than
107        // packet_remaining_bytes it might happen that max_first_fragment_size >=
108        // obu.size. Also, since checks above verified |obu| should not be put
109        // completely into the |packet|, leave at least 1 byte for later packet.
110        let first_fragment_size = min(obu.size - 1, max_first_fragment_size);
111        if first_fragment_size == 0 {
112            // Rather than writing 0-size element at the tail of the packet,
113            // 'uninsert' the |obu| from the |packet|.
114            packet.num_obu_elements -= 1;
115            packet.packet_size -= previous_obu_extra_size;
116        } else {
117            packet.packet_size += first_fragment_size;
118            if must_write_obu_element_size {
119                packet.packet_size += leb128_size(first_fragment_size as u32);
120            }
121            packet.last_obu_size = first_fragment_size;
122        }
123        packets.push(packet);
124
125        // Add middle fragments that occupy all of the packet.
126        // These are easy because
127        // - one obu per packet imply no need to store the size of the obu.
128        // - this packets are nor the first nor the last packets of the frame, so
129        // packet capacity is always limits.max_payload_len.
130        let mut obu_offset = first_fragment_size;
131        while obu_offset + max_payload_size < obu.size {
132            let mut packet = PacketMetadata::new(obu_index);
133            packet.num_obu_elements = 1;
134            packet.first_obu_offset = obu_offset;
135            let middle_fragment_size = max_payload_size;
136            packet.last_obu_size = middle_fragment_size;
137            packet.packet_size = middle_fragment_size;
138            packets.push(packet);
139            obu_offset += max_payload_size;
140        }
141
142        // Add the last fragment of the obu.
143        let mut last_fragment_size = obu.size - obu_offset;
144        // Check for corner case where last fragment of the last obu is too large
145        // to fit into last packet, but may fully fit into semi-last packet.
146        if is_last_obu && last_fragment_size > max_payload_size {
147            // Split last fragments into two.
148            // Try to even packet sizes rather than payload sizes across the last
149            // two packets.
150            let mut semi_last_fragment_size = last_fragment_size / 2;
151            // But leave at least one payload byte for the last packet to avoid
152            // weird scenarios where size of the fragment is zero and rtp payload has
153            // nothing except for an aggregation header.
154            if semi_last_fragment_size >= last_fragment_size {
155                semi_last_fragment_size = last_fragment_size - 1;
156            }
157            last_fragment_size -= semi_last_fragment_size;
158            let mut packet = PacketMetadata::new(obu_index);
159            packet.first_obu_offset = obu_offset;
160            packet.last_obu_size = semi_last_fragment_size;
161            packet.packet_size = semi_last_fragment_size;
162            packets.push(packet);
163            obu_offset += semi_last_fragment_size
164        }
165        let mut last_packet = PacketMetadata::new(obu_index);
166        last_packet.num_obu_elements = 1;
167        last_packet.first_obu_offset = obu_offset;
168        last_packet.last_obu_size = last_fragment_size;
169        last_packet.packet_size = last_fragment_size;
170        packets.push(last_packet);
171        packet_remaining_bytes = max_payload_size - last_fragment_size;
172    }
173
174    packets
175}

177/// Returns the aggregation header for the packet.
178/// Reference: https://aomediacodec.github.io/av1-rtp-spec/#44-av1-aggregation-header
179pub fn get_aggregation_header(obus: &[Obu], packets: &[PacketMetadata], packet_index: usize) -> u8 {
180    let packet = &packets[packet_index];
181    let mut header: u8 = 0;
182
183    // Set Z flag: first obu element is continuation of the previous OBU.
184    let first_obu_element_is_fragment = packet.first_obu_offset > 0;
185    if first_obu_element_is_fragment {
186        header |= 1 << 7;
187    }
188
189    // Set Y flag: last obu element will be continuated in the next packet.
190    let last_obu_offset = if packet.num_obu_elements == 1 {
191        packet.first_obu_offset
192    } else {
193        0
194    };
195    let last_obu_is_fragment = last_obu_offset + packet.last_obu_size
196        < obus[packet.first_obu_index + packet.num_obu_elements - 1].size;
197    if last_obu_is_fragment {
198        header |= 1 << 6;
199    }
200
201    // Set W field: number of obu elements in the packet (when not too large).
202    if packet.num_obu_elements <= MAX_NUM_OBUS_TO_OMIT_SIZE {
203        header |= (packet.num_obu_elements as u8) << 4;
204    }
205
206    // Set N flag: beginning of a new coded video sequence.
207    // Encoder may produce key frame without a sequence header, thus double check
208    // incoming frame includes the sequence header. Since Temporal delimiter is
209    // already filtered out, sequence header should be the first obu when present.
210    //
211    // TODO: This is technically incorrect, since sequence headers may be present in delta frames.
212    //      However, unlike the Chromium implementation: https://chromium.googlesource.com/external/webrtc/+/4e513346ec56c829b3a6010664998469fc237b35/modules/rtp_rtcp/source/rtp_packetizer_av1.cc#345,
213    //      we do not have direct access to the whether this is a keyframe or a delta frame.
214    //      Thus for now we assume that every frame that starts with a sequence header is a keyframe,
215    //      which is not always true. This is the best we can do for now until implementing
216    //      a proper frame type detection, perhaps by parsing the FRAME_HEADER OBUs according to
217    //      https://aomediacodec.github.io/av1-spec/#ordering-of-obus:
218    //              A new coded video sequence is defined to start at each temporal unit which
219    //              satisfies both of the following conditions:
220    //                  - A sequence header OBU appears before the first frame header.
221    //                  - The first frame header has frame_type equal to KEY_FRAME, show_frame equal
222    //                    to 1, show_existing_frame equal to 0, and temporal_id equal to 0.
223    if packet_index == 0 && obu_type(obus.first().unwrap().header) == OBU_TYPE_SEQUENCE_HEADER {
224        header |= 1 << 3;
225    }
226    header
227}

229/// Returns the number of additional bytes needed to store the previous OBU
230/// element if an additional OBU element is added to the packet.
231fn additional_bytes_for_previous_obu_element(packet: &PacketMetadata) -> usize {
232    if packet.packet_size == 0 || packet.num_obu_elements > MAX_NUM_OBUS_TO_OMIT_SIZE {
233        // Packet is still empty => no last OBU element, no need to reserve space for it.
234        //  OR
235        // There are so many obu elements in the packet, all of them must be
236        // prepended with the length field. That imply space for the length of the
237        // last obu element is already reserved.
238        0
239    } else {
240        leb128_size(packet.last_obu_size as u32)
241    }
242}

244/// Given |remaining_bytes| free bytes left in a packet, returns max size of an
245/// OBU fragment that can fit into the packet.
246/// i.e. MaxFragmentSize + Leb128Size(MaxFragmentSize) <= remaining_bytes.
247fn max_fragment_size(remaining_bytes: usize) -> usize {
248    if remaining_bytes <= 1 {
249        return 0;
250    }
251    let mut i = 1;
252    loop {
253        if remaining_bytes < (1 << (7 * i)) + i {
254            return remaining_bytes - i;
255        }
256        i += 1;
257    }
