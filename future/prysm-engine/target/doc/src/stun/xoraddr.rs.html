42fn safe_xor_bytes(dst: &mut [u8], a: &[u8], b: &[u8]) -> usize {
43    let mut n = a.len();
44    if b.len() < n {
45        n = b.len();
46    }
47    if dst.len() < n {
48        n = dst.len();
49    }
50    for i in 0..n {
51        dst[i] = a[i] ^ b[i];
52    }
53    n
54}

65/// XORMappedAddress implements XOR-MAPPED-ADDRESS attribute.
66///
67/// RFC 5389 Section 15.2
68pub struct XorMappedAddress {
69    pub ip: IpAddr,
70    pub port: u16,
71}
72
73impl Default for XorMappedAddress {
74    fn default() -> Self {
75        XorMappedAddress {
76            ip: IpAddr::V4(Ipv4Addr::from(0)),
77            port: 0,
78        }
79    }
80}

82impl fmt::Display for XorMappedAddress {
83    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
84        let family = match self.ip {
85            IpAddr::V4(_) => FAMILY_IPV4,
86            IpAddr::V6(_) => FAMILY_IPV6,
87        };
88        if family == FAMILY_IPV4 {
89            write!(f, "{}:{}", self.ip, self.port)
90        } else {
91            write!(f, "[{}]:{}", self.ip, self.port)
92        }
93    }
94}

96impl Setter for XorMappedAddress {

99    fn add_to(&self, m: &mut Message) -> Result<()> {

104impl Getter for XorMappedAddress {
105    /// get_from decodes XOR-MAPPED-ADDRESS attribute in message and returns
106    /// error if any. While decoding, a.IP is reused if possible and can be
107    /// rendered to invalid state (e.g. if a.IP was set to IPv6 and then
108    /// IPv4 value were decoded into it), be careful.
109    fn get_from(&mut self, m: &Message) -> Result<()> {
110        self.get_from_as(m, ATTR_XORMAPPED_ADDRESS)
111    }
112}

114impl XorMappedAddress {
115    /// add_to_as adds XOR-MAPPED-ADDRESS value to m as t attribute.
116    pub fn add_to_as(&self, m: &mut Message, t: AttrType) -> Result<()> {
117        let (family, ip_len, ip) = match self.ip {
118            IpAddr::V4(ipv4) => (FAMILY_IPV4, IPV4LEN, ipv4.octets().to_vec()),
119            IpAddr::V6(ipv6) => (FAMILY_IPV6, IPV6LEN, ipv6.octets().to_vec()),
120        };
121
122        let mut value = [0; 32 + 128];
123        //value[0] = 0 // first 8 bits are zeroes
124        let mut xor_value = vec![0; IPV6LEN];
125        xor_value[4..].copy_from_slice(&m.transaction_id.0);
126        xor_value[0..4].copy_from_slice(&MAGIC_COOKIE.to_be_bytes());
127        value[0..2].copy_from_slice(&family.to_be_bytes());
128        value[2..4].copy_from_slice(&(self.port ^ (MAGIC_COOKIE >> 16) as u16).to_be_bytes());
129        xor_bytes(&mut value[4..4 + ip_len], &ip, &xor_value);
130        m.add(t, &value[..4 + ip_len]);
131        Ok(())
132    }
133
134    /// get_from_as decodes XOR-MAPPED-ADDRESS attribute value in message
135    /// getting it as for t type.
136    pub fn get_from_as(&mut self, m: &Message, t: AttrType) -> Result<()> {
137        let v = m.get(t)?;
138        if v.len() <= 4 {
139            return Err(Error::ErrUnexpectedEof);
140        }
141
142        let family = u16::from_be_bytes([v[0], v[1]]);
143        if family != FAMILY_IPV6 && family != FAMILY_IPV4 {
144            return Err(Error::Other(format!("bad value {family}")));
145        }
146
147        check_overflow(
148            t,
149            v[4..].len(),
150            if family == FAMILY_IPV4 {
151                IPV4LEN
152            } else {
153                IPV6LEN
154            },
155        )?;
156        self.port = u16::from_be_bytes([v[2], v[3]]) ^ (MAGIC_COOKIE >> 16) as u16;
157        let mut xor_value = vec![0; 4 + TRANSACTION_ID_SIZE];
158        xor_value[0..4].copy_from_slice(&MAGIC_COOKIE.to_be_bytes());
159        xor_value[4..].copy_from_slice(&m.transaction_id.0);
160
161        if family == FAMILY_IPV6 {
162            let mut ip = [0; IPV6LEN];
163            xor_bytes(&mut ip, &v[4..], &xor_value);
164            self.ip = IpAddr::V6(Ipv6Addr::from(ip));
165        } else {
166            let mut ip = [0; IPV4LEN];
167            xor_bytes(&mut ip, &v[4..], &xor_value);
168            self.ip = IpAddr::V4(Ipv4Addr::from(ip));
169        };
170
171        Ok(())
172    }
173}
