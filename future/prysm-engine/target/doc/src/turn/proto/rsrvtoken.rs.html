4use stun::attributes::*;
5use stun::checks::*;
6use stun::message::*;

8/// `ReservationToken` represents `RESERVATION-TOKEN` attribute.
9///
10/// The `RESERVATION-TOKEN` attribute contains a token that uniquely
11/// identifies a relayed transport address being held in reserve by the
12/// server. The server includes this attribute in a success response to
13/// tell the client about the token, and the client includes this
14/// attribute in a subsequent Allocate request to request the server use
15/// that relayed transport address for the allocation.
16///
17/// [RFC 5766 Section 14.9](https://www.rfc-editor.org/rfc/rfc5766#section-14.9).
18#[derive(Debug, Default, PartialEq, Eq)]
19pub struct ReservationToken(pub Vec<u8>);
20
21const RESERVATION_TOKEN_SIZE: usize = 8; // 8 bytes
22
23impl Setter for ReservationToken {
24    /// Adds `RESERVATION-TOKEN` to message.
25    fn add_to(&self, m: &mut Message) -> Result<(), stun::Error> {
26        check_size(ATTR_RESERVATION_TOKEN, self.0.len(), RESERVATION_TOKEN_SIZE)?;
27        m.add(ATTR_RESERVATION_TOKEN, &self.0);
28        Ok(())
29    }
30}
31
32impl Getter for ReservationToken {
33    /// Decodes `RESERVATION-TOKEN` from message.
34    fn get_from(&mut self, m: &Message) -> Result<(), stun::Error> {
35        let v = m.get(ATTR_RESERVATION_TOKEN)?;
36        check_size(ATTR_RESERVATION_TOKEN, v.len(), RESERVATION_TOKEN_SIZE)?;
37        self.0 = v;
38        Ok(())
39    }
40}
