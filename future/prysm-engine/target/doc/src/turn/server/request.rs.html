122    async fn process_message_handler(&mut self, m: &Message) -> Result<()> {
123        if m.typ.class == CLASS_INDICATION {
124            match m.typ.method {
125                METHOD_SEND => self.handle_send_indication(m).await,
126                _ => Err(Error::ErrUnexpectedClass),
127            }
128        } else if m.typ.class == CLASS_REQUEST {
129            match m.typ.method {
130                METHOD_ALLOCATE => self.handle_allocate_request(m).await,
131                METHOD_REFRESH => self.handle_refresh_request(m).await,
132                METHOD_CREATE_PERMISSION => self.handle_create_permission_request(m).await,
133                METHOD_CHANNEL_BIND => self.handle_channel_bind_request(m).await,
134                METHOD_BINDING => self.handle_binding_request(m).await,
135                _ => Err(Error::ErrUnexpectedClass),
136            }
137        } else {
138            Err(Error::ErrUnexpectedClass)
139        }
140    }

142    pub(crate) async fn authenticate_request(
143        &mut self,
144        m: &Message,
145        calling_method: Method,
146    ) -> Result<Option<(Username, MessageIntegrity)>> {
147        if !m.contains(ATTR_MESSAGE_INTEGRITY) {
148            self.respond_with_nonce(m, calling_method, CODE_UNAUTHORIZED)
149                .await?;
150            return Ok(None);
151        }
152
153        let mut nonce_attr = Nonce::new(ATTR_NONCE, String::new());
154        let mut username_attr = Username::new(ATTR_USERNAME, String::new());
155        let mut realm_attr = Realm::new(ATTR_REALM, String::new());
156        let bad_request_msg = build_msg(
157            m.transaction_id,
158            MessageType::new(calling_method, CLASS_ERROR_RESPONSE),
159            vec![Box::new(ErrorCodeAttribute {
160                code: CODE_BAD_REQUEST,
161                reason: vec![],
162            })],
163        )?;
164
165        if let Err(err) = nonce_attr.get_from(m) {
166            build_and_send_err(&self.conn, self.src_addr, bad_request_msg, err.into()).await?;
167            return Ok(None);
168        }
169
170        let to_be_deleted = {
171            // Assert Nonce exists and is not expired
172            let mut nonces = self.nonces.lock().await;
173
174            let to_be_deleted = if let Some(nonce_creation_time) = nonces.get(&nonce_attr.text) {
175                Instant::now()
176                    .checked_duration_since(*nonce_creation_time)
177                    .unwrap_or_else(|| Duration::from_secs(0))
178                    >= NONCE_LIFETIME
179            } else {
180                true
181            };
182
183            if to_be_deleted {
184                nonces.remove(&nonce_attr.text);
185            }
186            to_be_deleted
187        };
188
189        if to_be_deleted {
190            self.respond_with_nonce(m, calling_method, CODE_STALE_NONCE)
191                .await?;
192            return Ok(None);
193        }
194
195        if let Err(err) = realm_attr.get_from(m) {
196            build_and_send_err(&self.conn, self.src_addr, bad_request_msg, err.into()).await?;
197            return Ok(None);
198        }
199        if let Err(err) = username_attr.get_from(m) {
200            build_and_send_err(&self.conn, self.src_addr, bad_request_msg, err.into()).await?;
201            return Ok(None);
202        }
203
204        let our_key = match self.auth_handler.auth_handle(
205            &username_attr.to_string(),
206            &realm_attr.to_string(),
207            self.src_addr,
208        ) {
209            Ok(key) => key,
210            Err(_) => {
211                build_and_send_err(
212                    &self.conn,
213                    self.src_addr,
214                    bad_request_msg,
215                    Error::ErrNoSuchUser,
216                )
217                .await?;
218                return Ok(None);
219            }
220        };
221
222        let mi = MessageIntegrity(our_key);
223        if let Err(err) = mi.check(&mut m.clone()) {
224            build_and_send_err(&self.conn, self.src_addr, bad_request_msg, err.into()).await?;
225            Ok(None)
226        } else {
227            Ok(Some((username_attr, mi)))
228        }
229    }

231    async fn respond_with_nonce(
232        &mut self,
233        m: &Message,
234        calling_method: Method,
235        response_code: ErrorCode,
236    ) -> Result<()> {
237        let nonce = build_nonce()?;
238
239        {
240            // Nonce has already been taken
241            let mut nonces = self.nonces.lock().await;
242            if nonces.contains_key(&nonce) {
243                return Err(Error::ErrDuplicatedNonce);
244            }
245            nonces.insert(nonce.clone(), Instant::now());
246        }
247
248        let msg = build_msg(
249            m.transaction_id,
250            MessageType::new(calling_method, CLASS_ERROR_RESPONSE),
251            vec![
252                Box::new(ErrorCodeAttribute {
253                    code: response_code,
254                    reason: vec![],
255                }),
256                Box::new(Nonce::new(ATTR_NONCE, nonce)),
257                Box::new(Realm::new(ATTR_REALM, self.realm.clone())),
258            ],
259        )?;
260
261        build_and_send(&self.conn, self.src_addr, msg).await
262    }

281    /// https://tools.ietf.org/html/rfc5766#section-6.2
282    pub(crate) async fn handle_allocate_request(&mut self, m: &Message) -> Result<()> {
283        log::debug!("received AllocateRequest from {}", self.src_addr);
284
285        // 1. The server MUST require that the request be authenticated.  This
286        //    authentication MUST be done using the long-term credential
287        //    mechanism of [https://tools.ietf.org/html/rfc5389#section-10.2.2]
288        //    unless the client and server agree to use another mechanism through
289        //    some procedure outside the scope of this document.
290        let (username, message_integrity) =
291            if let Some(mi) = self.authenticate_request(m, METHOD_ALLOCATE).await? {
292                mi
293            } else {
294                log::debug!("no MessageIntegrity");
295                return Ok(());
296            };
297
298        let five_tuple = FiveTuple {
299            src_addr: self.src_addr,
300            dst_addr: self.conn.local_addr()?,
301            protocol: PROTO_UDP,
302        };
303        let mut requested_port = 0;
304        let mut reservation_token = "".to_owned();
305        let mut use_ipv4 = true;
306
307        // 2. The server checks if the 5-tuple is currently in use by an
308        //    existing allocation.  If yes, the server rejects the request with
309        //    a 437 (Allocation Mismatch) error.
310        if self
311            .allocation_manager
312            .get_allocation(&five_tuple)
313            .await
314            .is_some()
315        {
316            let msg = build_msg(
317                m.transaction_id,
318                MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
319                vec![Box::new(ErrorCodeAttribute {
320                    code: CODE_ALLOC_MISMATCH,
321                    reason: vec![],
322                })],
323            )?;
324            return build_and_send_err(
325                &self.conn,
326                self.src_addr,
327                msg,
328                Error::ErrRelayAlreadyAllocatedForFiveTuple,
329            )
330            .await;
331        }
332
333        // 3. The server checks if the request contains a REQUESTED-TRANSPORT
334        //    attribute.  If the REQUESTED-TRANSPORT attribute is not included
335        //    or is malformed, the server rejects the request with a 400 (Bad
336        //    Request) error.  Otherwise, if the attribute is included but
337        //    specifies a protocol other that UDP, the server rejects the
338        //    request with a 442 (Unsupported Transport Protocol) error.
339        let mut requested_transport = RequestedTransport::default();
340        if let Err(err) = requested_transport.get_from(m) {
341            let bad_request_msg = build_msg(
342                m.transaction_id,
343                MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
344                vec![Box::new(ErrorCodeAttribute {
345                    code: CODE_BAD_REQUEST,
346                    reason: vec![],
347                })],
348            )?;
349            return build_and_send_err(&self.conn, self.src_addr, bad_request_msg, err.into())
350                .await;
351        } else if requested_transport.protocol != PROTO_UDP {
352            let msg = build_msg(
353                m.transaction_id,
354                MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
355                vec![Box::new(ErrorCodeAttribute {
356                    code: CODE_UNSUPPORTED_TRANS_PROTO,
357                    reason: vec![],
358                })],
359            )?;
360            return build_and_send_err(
361                &self.conn,
362                self.src_addr,
363                msg,
364                Error::ErrRequestedTransportMustBeUdp,
365            )
366            .await;
367        }
368
369        // 4. The request may contain a DONT-FRAGMENT attribute.  If it does,
370        //    but the server does not support sending UDP datagrams with the DF
371        //    bit set to 1 (see Section 12), then the server treats the DONT-
372        //    FRAGMENT attribute in the Allocate request as an unknown
373        //    comprehension-required attribute.
374        if m.contains(ATTR_DONT_FRAGMENT) {
375            let msg = build_msg(
376                m.transaction_id,
377                MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
378                vec![
379                    Box::new(ErrorCodeAttribute {
380                        code: CODE_UNKNOWN_ATTRIBUTE,
381                        reason: vec![],
382                    }),
383                    Box::new(UnknownAttributes(vec![ATTR_DONT_FRAGMENT])),
384                ],
385            )?;
386            return build_and_send_err(
387                &self.conn,
388                self.src_addr,
389                msg,
390                Error::ErrNoDontFragmentSupport,
391            )
392            .await;
393        }
394
395        // 5.  The server checks if the request contains a RESERVATION-TOKEN
396        //     attribute.  If yes, and the request also contains an EVEN-PORT
397        //     attribute, then the server rejects the request with a 400 (Bad
398        //     Request) error.  Otherwise, it checks to see if the token is
399        //     valid (i.e., the token is in range and has not expired and the
400        //     corresponding relayed transport address is still available).  If
401        //     the token is not valid for some reason, the server rejects the
402        //     request with a 508 (Insufficient Capacity) error.
403        let mut reservation_token_attr = ReservationToken::default();
404        let reservation_token_attr_result = reservation_token_attr.get_from(m);
405        if reservation_token_attr_result.is_ok() {
406            let mut even_port = EvenPort::default();
407            if even_port.get_from(m).is_ok() {
408                let bad_request_msg = build_msg(
409                    m.transaction_id,
410                    MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
411                    vec![Box::new(ErrorCodeAttribute {
412                        code: CODE_BAD_REQUEST,
413                        reason: vec![],
414                    })],
415                )?;
416                return build_and_send_err(
417                    &self.conn,
418                    self.src_addr,
419                    bad_request_msg,
420                    Error::ErrRequestWithReservationTokenAndEvenPort,
421                )
422                .await;
423            }
424        }
425
426        // RFC 6156, Section 4.2:
427        //
428        // If it contains both a RESERVATION-TOKEN and a
429        // REQUESTED-ADDRESS-FAMILY, the server replies with a 400
430        // (Bad Request) Allocate error response.
431        //
432        // 4.2.1.  Unsupported Address Family
433        // This document defines the following new error response code:
434        // 440 (Address Family not Supported):  The server does not support the
435        // address family requested by the client.
436        let mut req_family = RequestedAddressFamily::default();
437        match req_family.get_from(m) {
438            Err(err) => {
439                // Currently, the RequestedAddressFamily::get_from() function returns
440                // Err::Other only when it is an unsupported address family.
441                if let stun::Error::Other(_) = err {
442                    let addr_family_not_supported_msg = build_msg(
443                        m.transaction_id,
444                        MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
445                        vec![Box::new(ErrorCodeAttribute {
446                            code: CODE_ADDR_FAMILY_NOT_SUPPORTED,
447                            reason: vec![],
448                        })],
449                    )?;
450                    return build_and_send_err(
451                        &self.conn,
452                        self.src_addr,
453                        addr_family_not_supported_msg,
454                        Error::ErrInvalidRequestedFamilyValue,
455                    )
456                    .await;
457                }
458            }
459            Ok(()) => {
460                if reservation_token_attr_result.is_ok() {
461                    let bad_request_msg = build_msg(
462                        m.transaction_id,
463                        MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
464                        vec![Box::new(ErrorCodeAttribute {
465                            code: CODE_BAD_REQUEST,
466                            reason: vec![],
467                        })],
468                    )?;
469
470                    return build_and_send_err(
471                        &self.conn,
472                        self.src_addr,
473                        bad_request_msg,
474                        Error::ErrRequestWithReservationTokenAndReqAddressFamily,
475                    )
476                    .await;
477                }
478
479                if req_family == REQUESTED_FAMILY_IPV6 {
480                    use_ipv4 = false;
481                }
482            }
483        }
484
485        // 6. The server checks if the request contains an EVEN-PORT attribute.
486        //    If yes, then the server checks that it can satisfy the request
487        //    (i.e., can allocate a relayed transport address as described
488        //    below).  If the server cannot satisfy the request, then the
489        //    server rejects the request with a 508 (Insufficient Capacity)
490        //    error.
491        let mut even_port = EvenPort::default();
492        if even_port.get_from(m).is_ok() {
493            let mut random_port = 1;
494
495            while random_port % 2 != 0 {
496                random_port = match self.allocation_manager.get_random_even_port().await {
497                    Ok(port) => port,
498                    Err(err) => {
499                        let insufficient_capacity_msg = build_msg(
500                            m.transaction_id,
501                            MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
502                            vec![Box::new(ErrorCodeAttribute {
503                                code: CODE_INSUFFICIENT_CAPACITY,
504                                reason: vec![],
505                            })],
506                        )?;
507                        return build_and_send_err(
508                            &self.conn,
509                            self.src_addr,
510                            insufficient_capacity_msg,
511                            err,
512                        )
513                        .await;
514                    }
515                };
516            }
517
518            requested_port = random_port;
519            reservation_token = rand_seq(8);
520        }
521
522        // 7. At any point, the server MAY choose to reject the request with a
523        //    486 (Allocation Quota Reached) error if it feels the client is
524        //    trying to exceed some locally defined allocation quota.  The
525        //    server is free to define this allocation quota any way it wishes,
526        //    but SHOULD define it based on the username used to authenticate
527        //    the request, and not on the client's transport address.
528
529        // 8. Also at any point, the server MAY choose to reject the request
530        //    with a 300 (Try Alternate) error if it wishes to redirect the
531        //    client to a different server.  The use of this error code and
532        //    attribute follow the specification in [RFC5389].
533        let lifetime_duration = allocation_lifetime(m);
534        let a = match self
535            .allocation_manager
536            .create_allocation(
537                five_tuple,
538                Arc::clone(&self.conn),
539                requested_port,
540                lifetime_duration,
541                username,
542                use_ipv4,
543            )
544            .await
545        {
546            Ok(a) => a,
547            Err(err) => {
548                let insufficient_capacity_msg = build_msg(
549                    m.transaction_id,
550                    MessageType::new(METHOD_ALLOCATE, CLASS_ERROR_RESPONSE),
551                    vec![Box::new(ErrorCodeAttribute {
552                        code: CODE_INSUFFICIENT_CAPACITY,
553                        reason: vec![],
554                    })],
555                )?;
556                return build_and_send_err(
557                    &self.conn,
558                    self.src_addr,
559                    insufficient_capacity_msg,
560                    err,
561                )
562                .await;
563            }
564        };
565
566        // Once the allocation is created, the server replies with a success
567        // response.  The success response contains:
568        //   * An XOR-RELAYED-ADDRESS attribute containing the relayed transport
569        //     address.
570        //   * A LIFETIME attribute containing the current value of the time-to-
571        //     expiry timer.
572        //   * A RESERVATION-TOKEN attribute (if a second relayed transport
573        //     address was reserved).
574        //   * An XOR-MAPPED-ADDRESS attribute containing the client's IP address
575        //     and port (from the 5-tuple).
576
577        let (src_ip, src_port) = (self.src_addr.ip(), self.src_addr.port());
578        let relay_ip = a.relay_addr.ip();
579        let relay_port = a.relay_addr.port();
580
581        let msg = {
582            if !reservation_token.is_empty() {
583                self.allocation_manager
584                    .create_reservation(reservation_token.clone(), relay_port)
585                    .await;
586            }
587
588            let mut response_attrs: Vec<Box<dyn Setter>> = vec![
589                Box::new(RelayedAddress {
590                    ip: relay_ip,
591                    port: relay_port,
592                }),
593                Box::new(Lifetime(lifetime_duration)),
594                Box::new(XorMappedAddress {
595                    ip: src_ip,
596                    port: src_port,
597                }),
598            ];
599
600            if !reservation_token.is_empty() {
601                response_attrs.push(Box::new(ReservationToken(
602                    reservation_token.as_bytes().to_vec(),
603                )));
604            }
605
606            response_attrs.push(Box::new(message_integrity));
607            build_msg(
608                m.transaction_id,
609                MessageType::new(METHOD_ALLOCATE, CLASS_SUCCESS_RESPONSE),
610                response_attrs,
611            )?
612        };
613
614        build_and_send(&self.conn, self.src_addr, msg).await
615    }

617    pub(crate) async fn handle_refresh_request(&mut self, m: &Message) -> Result<()> {
618        log::debug!("received RefreshRequest from {}", self.src_addr);
619
620        let (_, message_integrity) =
621            if let Some(mi) = self.authenticate_request(m, METHOD_REFRESH).await? {
622                mi
623            } else {
624                log::debug!("no MessageIntegrity");
625                return Ok(());
626            };
627
628        let lifetime_duration = allocation_lifetime(m);
629        let five_tuple = FiveTuple {
630            src_addr: self.src_addr,
631            dst_addr: self.conn.local_addr()?,
632            protocol: PROTO_UDP,
633        };
634
635        if lifetime_duration != Duration::from_secs(0) {
636            let a = self.allocation_manager.get_allocation(&five_tuple).await;
637            if let Some(a) = a {
638                // If a server receives a Refresh Request with a REQUESTED-ADDRESS-FAMILY
639                // attribute, and the attribute's value doesn't match the address
640                // family of the allocation, the server MUST reply with a 443 (Peer
641                // Address Family Mismatch) Refresh error response. [RFC 6156, Section 5.2]
642                let mut req_family = RequestedAddressFamily::default();
643                if req_family.get_from(m).is_ok()
644                    && ((req_family == REQUESTED_FAMILY_IPV6 && !a.relay_addr.is_ipv6())
645                        || (req_family == REQUESTED_FAMILY_IPV4 && !a.relay_addr.is_ipv4()))
646                {
647                    let peer_address_family_mismatch_msg = build_msg(
648                        m.transaction_id,
649                        MessageType::new(METHOD_REFRESH, CLASS_ERROR_RESPONSE),
650                        vec![Box::new(ErrorCodeAttribute {
651                            code: CODE_PEER_ADDR_FAMILY_MISMATCH,
652                            reason: vec![],
653                        })],
654                    )?;
655                    return build_and_send_err(
656                        &self.conn,
657                        self.src_addr,
658                        peer_address_family_mismatch_msg,
659                        Error::ErrPeerAddressFamilyMismatch,
660                    )
661                    .await;
662                }
663                a.refresh(lifetime_duration).await;
664            } else {
665                return Err(Error::ErrNoAllocationFound);
666            }
667        } else {
668            self.allocation_manager.delete_allocation(&five_tuple).await;
669        }
670
671        let msg = build_msg(
672            m.transaction_id,
673            MessageType::new(METHOD_REFRESH, CLASS_SUCCESS_RESPONSE),
674            vec![
675                Box::new(Lifetime(lifetime_duration)),
676                Box::new(message_integrity),
677            ],
678        )?;
679
680        build_and_send(&self.conn, self.src_addr, msg).await
681    }

683    pub(crate) async fn handle_create_permission_request(&mut self, m: &Message) -> Result<()> {
684        log::debug!("received CreatePermission from {}", self.src_addr);
685
686        let a = self
687            .allocation_manager
688            .get_allocation(&FiveTuple {
689                src_addr: self.src_addr,
690                dst_addr: self.conn.local_addr()?,
691                protocol: PROTO_UDP,
692            })
693            .await;
694
695        if let Some(a) = a {
696            let (_, message_integrity) = if let Some(mi) = self
697                .authenticate_request(m, METHOD_CREATE_PERMISSION)
698                .await?
699            {
700                mi
701            } else {
702                log::debug!("no MessageIntegrity");
703                return Ok(());
704            };
705            let mut add_count = 0;
706
707            {
708                for attr in &m.attributes.0 {
709                    if attr.typ != ATTR_XOR_PEER_ADDRESS {
710                        continue;
711                    }
712
713                    let mut peer_address = PeerAddress::default();
714                    if peer_address.get_from(m).is_err() {
715                        add_count = 0;
716                        break;
717                    }
718
719                    // If an XOR-PEER-ADDRESS attribute contains an address of an address
720                    // family different than that of the relayed transport address for the
721                    // allocation, the server MUST generate an error response with the 443
722                    // (Peer Address Family Mismatch) response code. [RFC 6156, Section 6.2]
723                    if (peer_address.ip.is_ipv4() && !a.relay_addr.is_ipv4())
724                        || (peer_address.ip.is_ipv6() && !a.relay_addr.is_ipv6())
725                    {
726                        let peer_address_family_mismatch_msg = build_msg(
727                            m.transaction_id,
728                            MessageType::new(METHOD_CREATE_PERMISSION, CLASS_ERROR_RESPONSE),
729                            vec![Box::new(ErrorCodeAttribute {
730                                code: CODE_PEER_ADDR_FAMILY_MISMATCH,
731                                reason: vec![],
732                            })],
733                        )?;
734                        return build_and_send_err(
735                            &self.conn,
736                            self.src_addr,
737                            peer_address_family_mismatch_msg,
738                            Error::ErrPeerAddressFamilyMismatch,
739                        )
740                        .await;
741                    }
742
743                    log::debug!(
744                        "adding permission for {}",
745                        format!("{}:{}", peer_address.ip, peer_address.port)
746                    );
747
748                    a.add_permission(Permission::new(SocketAddr::new(
749                        peer_address.ip,
750                        peer_address.port,
751                    )))
752                    .await;
753                    add_count += 1;
754                }
755            }
756
757            let mut resp_class = CLASS_SUCCESS_RESPONSE;
758            if add_count == 0 {
759                resp_class = CLASS_ERROR_RESPONSE;
760            }
761
762            let msg = build_msg(
763                m.transaction_id,
764                MessageType::new(METHOD_CREATE_PERMISSION, resp_class),
765                vec![Box::new(message_integrity)],
766            )?;
767
768            build_and_send(&self.conn, self.src_addr, msg).await
769        } else {
770            Err(Error::ErrNoAllocationFound)
771        }
772    }

774    pub(crate) async fn handle_send_indication(&mut self, m: &Message) -> Result<()> {
775        log::debug!("received SendIndication from {}", self.src_addr);
776
777        let a = self
778            .allocation_manager
779            .get_allocation(&FiveTuple {
780                src_addr: self.src_addr,
781                dst_addr: self.conn.local_addr()?,
782                protocol: PROTO_UDP,
783            })
784            .await;
785
786        if let Some(a) = a {
787            let mut data_attr = Data::default();
788            data_attr.get_from(m)?;
789
790            let mut peer_address = PeerAddress::default();
791            peer_address.get_from(m)?;
792
793            let msg_dst = SocketAddr::new(peer_address.ip, peer_address.port);
794
795            let has_perm = a.has_permission(&msg_dst).await;
796            if !has_perm {
797                return Err(Error::ErrNoPermission);
798            }
799
800            let l = a.relay_socket.send_to(&data_attr.0, msg_dst).await?;
801            if l != data_attr.0.len() {
802                Err(Error::ErrShortWrite)
803            } else {
804                #[cfg(feature = "metrics")]
805                a.relayed_bytes
806                    .fetch_add(data_attr.0.len(), Ordering::AcqRel);
807
808                Ok(())
809            }
810        } else {
811            Err(Error::ErrNoAllocationFound)
812        }
813    }

815    pub(crate) async fn handle_channel_bind_request(&mut self, m: &Message) -> Result<()> {
816        log::debug!("received ChannelBindRequest from {}", self.src_addr);
817
818        let a = self
819            .allocation_manager
820            .get_allocation(&FiveTuple {
821                src_addr: self.src_addr,
822                dst_addr: self.conn.local_addr()?,
823                protocol: PROTO_UDP,
824            })
825            .await;
826
827        if let Some(a) = a {
828            let bad_request_msg = build_msg(
829                m.transaction_id,
830                MessageType::new(METHOD_CHANNEL_BIND, CLASS_ERROR_RESPONSE),
831                vec![Box::new(ErrorCodeAttribute {
832                    code: CODE_BAD_REQUEST,
833                    reason: vec![],
834                })],
835            )?;
836
837            let (_, message_integrity) =
838                if let Some(mi) = self.authenticate_request(m, METHOD_CHANNEL_BIND).await? {
839                    mi
840                } else {
841                    log::debug!("no MessageIntegrity");
842                    return Ok(());
843                };
844            let mut channel = ChannelNumber::default();
845            if let Err(err) = channel.get_from(m) {
846                return build_and_send_err(&self.conn, self.src_addr, bad_request_msg, err.into())
847                    .await;
848            }
849
850            let mut peer_addr = PeerAddress::default();
851            match peer_addr.get_from(m) {
852                Err(err) => {
853                    return build_and_send_err(
854                        &self.conn,
855                        self.src_addr,
856                        bad_request_msg,
857                        err.into(),
858                    )
859                    .await;
860                }
861                _ => {
862                    // If the XOR-PEER-ADDRESS attribute contains an address of an address
863                    // family different than that of the relayed transport address for the
864                    // allocation, the server MUST generate an error response with the 443
865                    // (Peer Address Family Mismatch) response code. [RFC 6156, Section 7.2]
866                    if (peer_addr.ip.is_ipv4() && !a.relay_addr.is_ipv4())
867                        || (peer_addr.ip.is_ipv6() && !a.relay_addr.is_ipv6())
868                    {
869                        let peer_address_family_mismatch_msg = build_msg(
870                            m.transaction_id,
871                            MessageType::new(METHOD_CHANNEL_BIND, CLASS_ERROR_RESPONSE),
872                            vec![Box::new(ErrorCodeAttribute {
873                                code: CODE_PEER_ADDR_FAMILY_MISMATCH,
874                                reason: vec![],
875                            })],
876                        )?;
877                        return build_and_send_err(
878                            &self.conn,
879                            self.src_addr,
880                            peer_address_family_mismatch_msg,
881                            Error::ErrPeerAddressFamilyMismatch,
882                        )
883                        .await;
884                    }
885                }
886            }
887
888            log::debug!(
889                "binding channel {} to {}",
890                channel,
891                format!("{}:{}", peer_addr.ip, peer_addr.port)
892            );
893
894            let result = {
895                a.add_channel_bind(
896                    ChannelBind::new(channel, SocketAddr::new(peer_addr.ip, peer_addr.port)),
897                    self.channel_bind_timeout,
898                )
899                .await
900            };
901            if let Err(err) = result {
902                return build_and_send_err(&self.conn, self.src_addr, bad_request_msg, err).await;
903            }
904
905            let msg = build_msg(
906                m.transaction_id,
907                MessageType::new(METHOD_CHANNEL_BIND, CLASS_SUCCESS_RESPONSE),
908                vec![Box::new(message_integrity)],
909            )?;
910            build_and_send(&self.conn, self.src_addr, msg).await
911        } else {
912            Err(Error::ErrNoAllocationFound)
913        }
914    }

916    pub(crate) async fn handle_channel_data(&mut self, c: &ChannelData) -> Result<()> {
917        log::debug!("received ChannelData from {}", self.src_addr);
918
919        let a = self
920            .allocation_manager
921            .get_allocation(&FiveTuple {
922                src_addr: self.src_addr,
923                dst_addr: self.conn.local_addr()?,
924                protocol: PROTO_UDP,
925            })
926            .await;
927
928        if let Some(a) = a {
929            let channel = a.get_channel_addr(&c.number).await;
930            if let Some(peer) = channel {
931                let l = a.relay_socket.send_to(&c.data, peer).await?;
932                if l != c.data.len() {
933                    Err(Error::ErrShortWrite)
934                } else {
935                    #[cfg(feature = "metrics")]
936                    a.relayed_bytes.fetch_add(c.data.len(), Ordering::AcqRel);
937
938                    Ok(())
939                }
940            } else {
941                Err(Error::ErrNoSuchChannelBind)
942            }
943        } else {
944            Err(Error::ErrNoAllocationFound)
945        }
946    }

949pub(crate) fn rand_seq(n: usize) -> String {
950    let letters = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ".as_bytes();
951    let mut buf = vec![0u8; n];
952    for b in &mut buf {
953        *b = letters[rand::random::<usize>() % letters.len()];
954    }
955    String::from_utf8(buf).unwrap_or_default()
956}

958pub(crate) fn build_nonce() -> Result<String> {
959    /* #nosec */
960    let mut s = String::new();
961    s.push_str(
962        format!(
963            "{}",
964            SystemTime::now()
965                .duration_since(SystemTime::UNIX_EPOCH)?
966                .as_nanos()
967        )
968        .as_str(),
969    );
970    s.push_str(format!("{}", rand::random::<u64>()).as_str());
971
972    let mut h = Md5::new();
973    h.update(s.as_bytes());
974    Ok(format!("{:x}", h.finalize()))
975}

998pub(crate) fn build_msg(
999    transaction_id: TransactionId,
1000    msg_type: MessageType,
1001    mut additional: Vec<Box<dyn Setter>>,
1002) -> Result<Message> {
1003    let mut attrs: Vec<Box<dyn Setter>> = vec![
1004        Box::new(Message {
1005            transaction_id,
1006            ..Default::default()
1007        }),
1008        Box::new(msg_type),
1009    ];
1010
1011    attrs.append(&mut additional);
1012
1013    let mut msg = Message::new();
1014    msg.build(&attrs)?;
1015    Ok(msg)
1016}
