121    /// state_change requires the caller holds the lock
122    async fn state_change(&self, state: RTCDtlsTransportState) {
123        self.state.store(state as u8, Ordering::SeqCst);
124        if let Some(handler) = &*self.on_state_change_handler.load() {
125            let mut f = handler.lock().await;
126            f(state).await;
127        }
128    }

133        self.on_state_change_handler
134            .store(Some(Arc::new(Mutex::new(f))));

142    /// write_rtcp sends a user provided RTCP packet to the connected peer. If no peer is connected the
143    /// packet is discarded.
144    pub async fn write_rtcp(
145        &self,
146        pkts: &[Box<dyn rtcp::packet::Packet + Send + Sync>],
147    ) -> Result<usize> {
148        let srtcp_session = self.srtcp_session.lock().await;
149        if let Some(srtcp_session) = &*srtcp_session {
150            let raw = rtcp::packet::marshal(pkts)?;
151            Ok(srtcp_session.write(&raw, false).await?)
152        } else {
153            Ok(0)
154        }
155    }

178    pub(crate) async fn start_srtp(&self) -> Result<()> {
179        let profile = {
180            let srtp_protection_profile = self.srtp_protection_profile.lock().await;
181            *srtp_protection_profile
182        };
183
184        let mut srtp_config = srtp::config::Config {
185            profile,
186            ..Default::default()
187        };
188
189        if self.setting_engine.replay_protection.srtp != 0 {
190            srtp_config.remote_rtp_options = Some(srtp::option::srtp_replay_protection(
191                self.setting_engine.replay_protection.srtp,
192            ));
193        } else if self.setting_engine.disable_srtp_replay_protection {
194            srtp_config.remote_rtp_options = Some(srtp::option::srtp_no_replay_protection());
195        }
196
197        if let Some(conn) = self.conn().await {
198            let conn_state = conn.connection_state().await;
199            srtp_config
200                .extract_session_keys_from_dtls(conn_state, self.role().await == DTLSRole::Client)
201                .await?;
202        } else {
203            return Err(Error::ErrDtlsTransportNotStarted);
204        }
205
206        {
207            let mut srtp_session = self.srtp_session.lock().await;
208            *srtp_session = {
209                let se = self.srtp_endpoint.lock().await;
210                if let Some(srtp_endpoint) = &*se {
211                    Some(Arc::new(
212                        Session::new(
213                            Arc::clone(srtp_endpoint) as Arc<dyn Conn + Send + Sync>,
214                            srtp_config,
215                            true,
216                        )
217                        .await?,
218                    ))
219                } else {
220                    None
221                }
222            };
223        }
224
225        let mut srtcp_config = srtp::config::Config {
226            profile,
227            ..Default::default()
228        };
229        if self.setting_engine.replay_protection.srtcp != 0 {
230            srtcp_config.remote_rtcp_options = Some(srtp::option::srtcp_replay_protection(
231                self.setting_engine.replay_protection.srtcp,
232            ));
233        } else if self.setting_engine.disable_srtcp_replay_protection {
234            srtcp_config.remote_rtcp_options = Some(srtp::option::srtcp_no_replay_protection());
235        }
236
237        if let Some(conn) = self.conn().await {
238            let conn_state = conn.connection_state().await;
239            srtcp_config
240                .extract_session_keys_from_dtls(conn_state, self.role().await == DTLSRole::Client)
241                .await?;
242        } else {
243            return Err(Error::ErrDtlsTransportNotStarted);
244        }
245
246        {
247            let mut srtcp_session = self.srtcp_session.lock().await;
248            *srtcp_session = {
249                let se = self.srtcp_endpoint.lock().await;
250                if let Some(srtcp_endpoint) = &*se {
251                    Some(Arc::new(
252                        Session::new(
253                            Arc::clone(srtcp_endpoint) as Arc<dyn Conn + Send + Sync>,
254                            srtcp_config,
255                            false,
256                        )
257                        .await?,
258                    ))
259                } else {
260                    None
261                }
262            };
263        }
264
265        {
266            let mut srtp_ready_tx = self.srtp_ready_tx.lock().await;
267            srtp_ready_tx.take();
268            if srtp_ready_tx.is_none() {
269                self.srtp_ready_signal.store(true, Ordering::SeqCst);
270            }
271        }
272
273        Ok(())
274    }

286    pub(crate) async fn role(&self) -> DTLSRole {
287        // If remote has an explicit role use the inverse
288        {
289            let remote_parameters = self.remote_parameters.lock().await;
290            match remote_parameters.role {
291                DTLSRole::Client => return DTLSRole::Server,
292                DTLSRole::Server => return DTLSRole::Client,
293                _ => {}
294            };
295        }
296
297        // If SettingEngine has an explicit role
298        match self.setting_engine.answering_dtls_role {
299            DTLSRole::Server => return DTLSRole::Server,
300            DTLSRole::Client => return DTLSRole::Client,
301            _ => {}
302        };
303
304        // Remote was auto and no explicit role was configured via SettingEngine
305        if self.ice_transport.role().await == RTCIceRole::Controlling {
306            return DTLSRole::Server;
307        }
308
309        DEFAULT_DTLS_ROLE_ANSWER
310    }

318    async fn prepare_transport(
319        &self,
320        remote_parameters: DTLSParameters,
321    ) -> Result<(DTLSRole, dtls::config::Config)> {
322        self.ensure_ice_conn()?;
323
324        if self.state() != RTCDtlsTransportState::New {
325            return Err(Error::ErrInvalidDTLSStart);
326        }
327
328        {
329            let mut srtp_endpoint = self.srtp_endpoint.lock().await;
330            *srtp_endpoint = self.ice_transport.new_endpoint(Box::new(match_srtp)).await;
331        }
332        {
333            let mut srtcp_endpoint = self.srtcp_endpoint.lock().await;
334            *srtcp_endpoint = self.ice_transport.new_endpoint(Box::new(match_srtcp)).await;
335        }
336        {
337            let mut rp = self.remote_parameters.lock().await;
338            *rp = remote_parameters;
339        }
340
341        let certificate = if let Some(cert) = self.certificates.first() {
342            cert.dtls_certificate.clone()
343        } else {
344            return Err(Error::ErrNonCertificate);
345        };
346        self.state_change(RTCDtlsTransportState::Connecting).await;
347
348        Ok((
349            self.role().await,
350            dtls::config::Config {
351                certificates: vec![certificate],
352                srtp_protection_profiles: if !self
353                    .setting_engine
354                    .srtp_protection_profiles
355                    .is_empty()
356                {
357                    self.setting_engine.srtp_protection_profiles.clone()
358                } else {
359                    default_srtp_protection_profiles()
360                },
361                client_auth: ClientAuthType::RequireAnyClientCert,
362                insecure_skip_verify: true,
363                insecure_verification: self.setting_engine.allow_insecure_verification_algorithm,
364                ..Default::default()
365            },
366        ))
367    }

369    /// start DTLS transport negotiation with the parameters of the remote DTLS transport
370    pub async fn start(&self, remote_parameters: DTLSParameters) -> Result<()> {
371        let dtls_conn_result = if let Some(dtls_endpoint) =
372            self.ice_transport.new_endpoint(Box::new(match_dtls)).await
373        {
374            let (role, mut dtls_config) = self.prepare_transport(remote_parameters).await?;
375            if self.setting_engine.replay_protection.dtls != 0 {
376                dtls_config.replay_protection_window = self.setting_engine.replay_protection.dtls;
377            }
378
379            // Connect as DTLS Client/Server, function is blocking and we
380            // must not hold the DTLSTransport lock
381            if role == DTLSRole::Client {
382                dtls::conn::DTLSConn::new(
383                    dtls_endpoint as Arc<dyn Conn + Send + Sync>,
384                    dtls_config,
385                    true,
386                    None,
387                )
388                .await
389            } else {
390                dtls::conn::DTLSConn::new(
391                    dtls_endpoint as Arc<dyn Conn + Send + Sync>,
392                    dtls_config,
393                    false,
394                    None,
395                )
396                .await
397            }
398        } else {
399            Err(dtls::Error::Other(
400                "ice_transport.new_endpoint failed".to_owned(),
401            ))
402        };
403
404        let dtls_conn = match dtls_conn_result {
405            Ok(dtls_conn) => dtls_conn,
406            Err(err) => {
407                self.state_change(RTCDtlsTransportState::Failed).await;
408                return Err(err.into());
409            }
410        };
411
412        let srtp_profile = dtls_conn.selected_srtpprotection_profile();
413        {
414            let mut srtp_protection_profile = self.srtp_protection_profile.lock().await;
415            *srtp_protection_profile = match srtp_profile {
416                dtls::extension::extension_use_srtp::SrtpProtectionProfile::Srtp_Aead_Aes_128_Gcm => {
417                    srtp::protection_profile::ProtectionProfile::AeadAes128Gcm
418                }
419                dtls::extension::extension_use_srtp::SrtpProtectionProfile::Srtp_Aead_Aes_256_Gcm => {
420                    srtp::protection_profile::ProtectionProfile::AeadAes256Gcm
421                }
422                dtls::extension::extension_use_srtp::SrtpProtectionProfile::Srtp_Aes128_Cm_Hmac_Sha1_80 => {
423                    srtp::protection_profile::ProtectionProfile::Aes128CmHmacSha1_80
424                }
425                dtls::extension::extension_use_srtp::SrtpProtectionProfile::Srtp_Aes128_Cm_Hmac_Sha1_32 => {
426                    srtp::protection_profile::ProtectionProfile::Aes128CmHmacSha1_32
427                }
428                _ => {
429                    if let Err(err) = dtls_conn.close().await {
430                        log::error!("{}", err);
431                    }
432
433                    self.state_change(RTCDtlsTransportState::Failed).await;
434                    return Err(Error::ErrNoSRTPProtectionProfile);
435                }
436            };
437        }
438
439        // Check the fingerprint if a certificate was exchanged
440        let remote_certs = &dtls_conn.connection_state().await.peer_certificates;
441        if remote_certs.is_empty() {
442            if let Err(err) = dtls_conn.close().await {
443                log::error!("{}", err);
444            }
445
446            self.state_change(RTCDtlsTransportState::Failed).await;
447            return Err(Error::ErrNoRemoteCertificate);
448        }
449
450        {
451            let mut remote_certificate = self.remote_certificate.lock().await;
452            *remote_certificate = Bytes::from(remote_certs[0].clone());
453        }
454
455        if !self
456            .setting_engine
457            .disable_certificate_fingerprint_verification
458        {
459            if let Err(err) = self.validate_fingerprint(&remote_certs[0]).await {
460                if let Err(close_err) = dtls_conn.close().await {
461                    log::error!("{}", close_err);
462                }
463
464                self.state_change(RTCDtlsTransportState::Failed).await;
465                return Err(err);
466            }
467        }
468
469        {
470            let mut conn = self.conn.lock().await;
471            *conn = Some(Arc::new(dtls_conn));
472        }
473        self.state_change(RTCDtlsTransportState::Connected).await;
474
475        self.start_srtp().await
476    }

478    /// stops and closes the DTLSTransport object.
479    pub async fn stop(&self) -> Result<()> {
480        // Try closing everything and collect the errors
481        let mut close_errs: Vec<Error> = vec![];
482        {
483            let srtp_session = {
484                let mut srtp_session = self.srtp_session.lock().await;
485                srtp_session.take()
486            };
487            if let Some(srtp_session) = srtp_session {
488                match srtp_session.close().await {
489                    Ok(_) => {}
490                    Err(err) => {
491                        close_errs.push(err.into());
492                    }
493                };
494            }
495        }
496
497        {
498            let srtcp_session = {
499                let mut srtcp_session = self.srtcp_session.lock().await;
500                srtcp_session.take()
501            };
502            if let Some(srtcp_session) = srtcp_session {
503                match srtcp_session.close().await {
504                    Ok(_) => {}
505                    Err(err) => {
506                        close_errs.push(err.into());
507                    }
508                };
509            }
510        }
511
512        {
513            let simulcast_streams: Vec<Arc<Stream>> = {
514                let mut simulcast_streams = self.simulcast_streams.lock().await;
515                simulcast_streams.drain().map(|(_, v)| v).collect()
516            };
517            for ss in simulcast_streams {
518                match ss.close().await {
519                    Ok(_) => {}
520                    Err(err) => {
521                        close_errs.push(Error::new(format!(
522                            "simulcast_streams ssrc={}: {}",
523                            ss.get_ssrc(),
524                            err
525                        )));
526                    }
527                };
528            }
529        }
530
531        if let Some(conn) = self.conn().await {
532            // dtls_transport connection may be closed on sctp close.
533            match conn.close().await {
534                Ok(_) => {}
535                Err(err) => {
536                    if err.to_string() != dtls::Error::ErrConnClosed.to_string() {
537                        close_errs.push(err.into());
538                    }
539                }
540            }
541        }
542
543        self.state_change(RTCDtlsTransportState::Closed).await;
544
545        flatten_errs(close_errs)
546    }

548    pub(crate) async fn validate_fingerprint(&self, remote_cert: &[u8]) -> Result<()> {
549        let remote_parameters = self.remote_parameters.lock().await;
550        for fp in &remote_parameters.fingerprints {
551            if fp.algorithm != "sha-256" {
552                return Err(Error::ErrUnsupportedFingerprintAlgorithm);
553            }
554
555            let mut h = Sha256::new();
556            h.update(remote_cert);
557            let hashed = h.finalize();
558            let values: Vec<String> = hashed.iter().map(|x| format! {"{x:02x}"}).collect();
559            let remote_value = values.join(":").to_lowercase();
560
561            if remote_value == fp.value.to_lowercase() {
562                return Ok(());
563            }
564        }
565
566        Err(Error::ErrNoMatchingCertificateFingerprint)
567    }

587    pub(crate) async fn streams_for_ssrc(
588        &self,
589        ssrc: SSRC,
590        stream_info: &StreamInfo,
591        interceptor: &Arc<dyn Interceptor + Send + Sync>,
592    ) -> Result<(
593        Arc<srtp::stream::Stream>,
594        Arc<dyn RTPReader + Send + Sync>,
595        Arc<srtp::stream::Stream>,
596        Arc<dyn RTCPReader + Send + Sync>,
597    )> {
598        let srtp_session = self
599            .get_srtp_session()
600            .await
601            .ok_or(Error::ErrDtlsTransportNotStarted)?;
602        //log::debug!("streams_for_ssrc: srtp_session.listen ssrc={}", ssrc);
603        let rtp_read_stream = srtp_session.open(ssrc).await;
604        let rtp_stream_reader = Arc::clone(&rtp_read_stream) as Arc<dyn RTPReader + Send + Sync>;
605        let rtp_interceptor = interceptor
606            .bind_remote_stream(stream_info, rtp_stream_reader)
607            .await;
608
609        let srtcp_session = self
610            .get_srtcp_session()
611            .await
612            .ok_or(Error::ErrDtlsTransportNotStarted)?;
613        //log::debug!("streams_for_ssrc: srtcp_session.listen ssrc={}", ssrc);
614        let rtcp_read_stream = srtcp_session.open(ssrc).await;
615        let rtcp_stream_reader = Arc::clone(&rtcp_read_stream) as Arc<dyn RTCPReader + Send + Sync>;
616        let rtcp_interceptor = interceptor.bind_rtcp_reader(rtcp_stream_reader).await;
617
618        Ok((
619            rtp_read_stream,
620            rtp_interceptor,
621            rtcp_read_stream,
622            rtcp_interceptor,
623        ))
624    }
