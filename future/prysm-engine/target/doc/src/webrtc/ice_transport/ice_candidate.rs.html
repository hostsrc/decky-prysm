16/// ICECandidate represents a ice candidate
17///
18/// ## Specifications
19///
20/// * [MDN]
21/// * [W3C]
22///
23/// [MDN]: https://developer.mozilla.org/en-US/docs/Web/API/RTCIceCandidate
24/// [W3C]: https://w3c.github.io/webrtc-pc/#rtcicecandidate-interface
25#[derive(Default, Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
26pub struct RTCIceCandidate {
27    pub stats_id: String,
28    pub foundation: String,
29    pub priority: u32,
30    pub address: String,
31    pub protocol: RTCIceProtocol,
32    pub port: u16,
33    pub typ: RTCIceCandidateType,
34    pub component: u16,
35    pub related_address: String,
36    pub related_port: u16,
37    pub tcp_type: String,
38}

40/// Conversion for ice_candidates
41pub(crate) fn rtc_ice_candidates_from_ice_candidates(
42    ice_candidates: &[Arc<dyn Candidate + Send + Sync>],
43) -> Vec<RTCIceCandidate> {
44    ice_candidates.iter().map(|c| c.into()).collect()
45}

47impl From<&Arc<dyn Candidate + Send + Sync>> for RTCIceCandidate {
48    fn from(c: &Arc<dyn Candidate + Send + Sync>) -> Self {
49        let typ: RTCIceCandidateType = c.candidate_type().into();
50        let protocol = RTCIceProtocol::from(c.network_type().network_short().as_str());
51        let (related_address, related_port) = if let Some(ra) = c.related_address() {
52            (ra.address, ra.port)
53        } else {
54            (String::new(), 0)
55        };
56
57        RTCIceCandidate {
58            stats_id: c.id(),
59            foundation: c.foundation(),
60            priority: c.priority(),
61            address: c.address(),
62            protocol,
63            port: c.port(),
64            component: c.component(),
65            typ,
66            tcp_type: c.tcp_type().to_string(),
67            related_address,
68            related_port,
69        }
70    }
71}

73impl RTCIceCandidate {
74    pub(crate) fn to_ice(&self) -> Result<impl Candidate> {
75        let candidate_id = self.stats_id.clone();
76        let base_config = CandidateBaseConfig {
77            candidate_id,
78            network: self.protocol.to_string(),
79            address: self.address.clone(),
80            port: self.port,
81            component: self.component,
82            //tcp_type: ice.NewTCPType(c.TCPType),
83            foundation: self.foundation.clone(),
84            priority: self.priority,
85            ..Default::default()
86        };
87
88        let c = match self.typ {
89            RTCIceCandidateType::Host => {
90                let config = CandidateHostConfig {
91                    base_config,
92                    ..Default::default()
93                };
94                config.new_candidate_host()?
95            }
96            RTCIceCandidateType::Srflx => {
97                let config = CandidateServerReflexiveConfig {
98                    base_config,
99                    rel_addr: self.related_address.clone(),
100                    rel_port: self.related_port,
101                };
102                config.new_candidate_server_reflexive()?
103            }
104            RTCIceCandidateType::Prflx => {
105                let config = CandidatePeerReflexiveConfig {
106                    base_config,
107                    rel_addr: self.related_address.clone(),
108                    rel_port: self.related_port,
109                };
110                config.new_candidate_peer_reflexive()?
111            }
112            RTCIceCandidateType::Relay => {
113                let config = CandidateRelayConfig {
114                    base_config,
115                    rel_addr: self.related_address.clone(),
116                    rel_port: self.related_port,
117                    relay_client: None, //TODO?
118                };
119                config.new_candidate_relay()?
120            }
121            _ => return Err(Error::ErrICECandidateTypeUnknown),
122        };
123
124        Ok(c)
125    }
126
127    /// to_json returns an ICECandidateInit
128    /// as indicated by the spec <https://w3c.github.io/webrtc-pc/#dom-rtcicecandidate-tojson>
129    pub fn to_json(&self) -> Result<RTCIceCandidateInit> {
130        let candidate = self.to_ice()?;
131
132        Ok(RTCIceCandidateInit {
133            candidate: format!("candidate:{}", candidate.marshal()),
134            sdp_mid: Some("".to_owned()),
135            sdp_mline_index: Some(0u16),
136            username_fragment: None,
137        })
138    }
139}

162#[cfg(test)]
163mod test {
164    use super::*;
165
166    #[test]
167    fn test_ice_candidate_serialization() {
168        let tests = vec![
169            (
170                RTCIceCandidateInit {
171                    candidate: "candidate:abc123".to_string(),
172                    sdp_mid: Some("0".to_string()),
173                    sdp_mline_index: Some(0),
174                    username_fragment: Some("def".to_string()),
175                },
176                r#"{"candidate":"candidate:abc123","sdpMid":"0","sdpMLineIndex":0,"usernameFragment":"def"}"#,
177            ),
178            (
179                RTCIceCandidateInit {
180                    candidate: "candidate:abc123".to_string(),
181                    sdp_mid: None,
182                    sdp_mline_index: None,
183                    username_fragment: None,
184                },
185                r#"{"candidate":"candidate:abc123","sdpMid":null,"sdpMLineIndex":null,"usernameFragment":null}"#,
186            ),
187        ];
188
189        for (candidate_init, expected_string) in tests {
190            let result = serde_json::to_string(&candidate_init);
191            assert!(result.is_ok(), "testCase: marshal err: {result:?}");
192            let candidate_data = result.unwrap();
193            assert_eq!(candidate_data, expected_string, "string is not expected");
194
195            let result = serde_json::from_str::<RTCIceCandidateInit>(&candidate_data);
196            assert!(result.is_ok(), "testCase: unmarshal err: {result:?}");
197            if let Ok(actual_candidate_init) = result {
198                assert_eq!(actual_candidate_init, candidate_init);
199            }
200        }
201    }
