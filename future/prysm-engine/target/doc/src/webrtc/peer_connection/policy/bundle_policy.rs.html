5/// BundlePolicy affects which media tracks are negotiated if the remote
6/// endpoint is not bundle-aware, and what ICE candidates are gathered. If the
7/// remote endpoint is bundle-aware, all media tracks and data channels are
8/// bundled onto the same transport.
9///
10/// ## Specifications
11///
12/// * [W3C]
13///
14/// [W3C]: https://w3c.github.io/webrtc-pc/#rtcbundlepolicy-enum
15#[derive(Default, Debug, PartialEq, Eq, Copy, Clone, Serialize, Deserialize)]
16pub enum RTCBundlePolicy {
17    #[default]
18    Unspecified = 0,
19
20    /// BundlePolicyBalanced indicates to gather ICE candidates for each
21    /// media type in use (audio, video, and data). If the remote endpoint is
22    /// not bundle-aware, negotiate only one audio and video track on separate
23    /// transports.
24    #[serde(rename = "balanced")]
25    Balanced = 1,
26
27    /// BundlePolicyMaxCompat indicates to gather ICE candidates for each
28    /// track. If the remote endpoint is not bundle-aware, negotiate all media
29    /// tracks on separate transports.
30    #[serde(rename = "max-compat")]
31    MaxCompat = 2,
32
33    /// BundlePolicyMaxBundle indicates to gather ICE candidates for only
34    /// one track. If the remote endpoint is not bundle-aware, negotiate only
35    /// one media track.
36    #[serde(rename = "max-bundle")]
37    MaxBundle = 3,
38}
39
40/// This is done this way because of a linter.
41const BUNDLE_POLICY_BALANCED_STR: &str = "balanced";
42const BUNDLE_POLICY_MAX_COMPAT_STR: &str = "max-compat";
43const BUNDLE_POLICY_MAX_BUNDLE_STR: &str = "max-bundle";
44
45impl From<&str> for RTCBundlePolicy {
46    /// NewSchemeType defines a procedure for creating a new SchemeType from a raw
47    /// string naming the scheme type.
48    fn from(raw: &str) -> Self {
49        match raw {
50            BUNDLE_POLICY_BALANCED_STR => RTCBundlePolicy::Balanced,
51            BUNDLE_POLICY_MAX_COMPAT_STR => RTCBundlePolicy::MaxCompat,
52            BUNDLE_POLICY_MAX_BUNDLE_STR => RTCBundlePolicy::MaxBundle,
53            _ => RTCBundlePolicy::Unspecified,
54        }
55    }
56}
57
58impl fmt::Display for RTCBundlePolicy {
59    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
60        match *self {
61            RTCBundlePolicy::Balanced => write!(f, "{BUNDLE_POLICY_BALANCED_STR}"),
62            RTCBundlePolicy::MaxCompat => write!(f, "{BUNDLE_POLICY_MAX_COMPAT_STR}"),
63            RTCBundlePolicy::MaxBundle => write!(f, "{BUNDLE_POLICY_MAX_BUNDLE_STR}"),
64            _ => write!(f, "{}", crate::UNSPECIFIED_STR),
65        }
66    }
67}
