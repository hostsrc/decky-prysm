136    /// Start the SCTPTransport. Since both local and remote parties must mutually
137    /// create an SCTPTransport, SCTP SO (Simultaneous Open) is used to establish
138    /// a connection over SCTP.
139    pub async fn start(&self, _remote_caps: SCTPTransportCapabilities) -> Result<()> {
140        if self.is_started.load(Ordering::SeqCst) {
141            return Ok(());
142        }
143        self.is_started.store(true, Ordering::SeqCst);
144
145        let dtls_transport = self.transport();
146        if let Some(net_conn) = &dtls_transport.conn().await {
147            let sctp_association = loop {
148                tokio::select! {
149                    _ = self.notify_tx.notified() => {
150                        // It seems like notify_tx is only notified on Stop so perhaps this check
151                        // is redundant.
152                        // TODO: Consider renaming notify_tx to shutdown_tx.
153                        if self.state.load(Ordering::SeqCst) == RTCSctpTransportState::Closed as u8 {
154                            return Err(Error::ErrSCTPTransportDTLS);
155                        }
156                    },
157                    association = sctp::association::Association::client(sctp::association::Config {
158                        net_conn: Arc::clone(net_conn) as Arc<dyn Conn + Send + Sync>,
159                        max_receive_buffer_size: 0,
160                        max_message_size: 0,
161                        name: String::new(),
162                    }) => {
163                        break Arc::new(association?);
164                    }
165                };
166            };
167
168            {
169                let mut sa = self.sctp_association.lock().await;
170                *sa = Some(Arc::clone(&sctp_association));
171            }
172            self.state
173                .store(RTCSctpTransportState::Connected as u8, Ordering::SeqCst);
174
175            let param = AcceptDataChannelParams {
176                notify_rx: self.notify_tx.clone(),
177                sctp_association,
178                data_channels: Arc::clone(&self.data_channels),
179                on_error_handler: Arc::clone(&self.on_error_handler),
180                on_data_channel_handler: Arc::clone(&self.on_data_channel_handler),
181                on_data_channel_opened_handler: Arc::clone(&self.on_data_channel_opened_handler),
182                data_channels_opened: Arc::clone(&self.data_channels_opened),
183                data_channels_accepted: Arc::clone(&self.data_channels_accepted),
184                setting_engine: Arc::clone(&self.setting_engine),
185            };
186            tokio::spawn(async move {
187                RTCSctpTransport::accept_data_channels(param).await;
188            });
189
190            Ok(())
191        } else {
192            Err(Error::ErrSCTPTransportDTLS)
193        }
194    }

196    /// Stop stops the SCTPTransport
197    pub async fn stop(&self) -> Result<()> {
198        {
199            let mut sctp_association = self.sctp_association.lock().await;
200            if let Some(sa) = sctp_association.take() {
201                sa.close().await?;
202            }
203        }
204
205        self.state
206            .store(RTCSctpTransportState::Closed as u8, Ordering::SeqCst);
207
208        self.notify_tx.notify_waiters();
209
210        Ok(())
211    }

213    async fn accept_data_channels(param: AcceptDataChannelParams) {
214        let dcs = param.data_channels.lock().await;
215        let mut existing_data_channels = Vec::new();
216        for dc in dcs.iter() {
217            if let Some(dc) = dc.data_channel.lock().await.clone() {
218                existing_data_channels.push(dc);
219            }
220        }
221        drop(dcs);
222
223        loop {
224            let dc = tokio::select! {
225                _ = param.notify_rx.notified() => break,
226                result = DataChannel::accept(
227                    &param.sctp_association,
228                    data::data_channel::Config::default(),
229                    &existing_data_channels,
230                ) => {
231                    match result {
232                        Ok(dc) => dc,
233                        Err(err) => {
234                            if data::Error::ErrStreamClosed == err {
235                                log::error!("Failed to accept data channel: {}", err);
236                                if let Some(handler) = &*param.on_error_handler.load() {
237                                    let mut f = handler.lock().await;
238                                    f(err.into()).await;
239                                }
240                            }
241                            break;
242                        }
243                    }
244                }
245            };
246
247            let mut max_retransmits = None;
248            let mut max_packet_life_time = None;
249            let val = dc.config.reliability_parameter as u16;
250            let ordered;
251
252            match dc.config.channel_type {
253                ChannelType::Reliable => {
254                    ordered = true;
255                }
256                ChannelType::ReliableUnordered => {
257                    ordered = false;
258                }
259                ChannelType::PartialReliableRexmit => {
260                    ordered = true;
261                    max_retransmits = Some(val);
262                }
263                ChannelType::PartialReliableRexmitUnordered => {
264                    ordered = false;
265                    max_retransmits = Some(val);
266                }
267                ChannelType::PartialReliableTimed => {
268                    ordered = true;
269                    max_packet_life_time = Some(val);
270                }
271                ChannelType::PartialReliableTimedUnordered => {
272                    ordered = false;
273                    max_packet_life_time = Some(val);
274                }
275            };
276
277            let negotiated = if dc.config.negotiated {
278                Some(dc.stream_identifier())
279            } else {
280                None
281            };
282            let rtc_dc = Arc::new(RTCDataChannel::new(
283                DataChannelParameters {
284                    label: dc.config.label.clone(),
285                    protocol: dc.config.protocol.clone(),
286                    negotiated,
287                    ordered,
288                    max_packet_life_time,
289                    max_retransmits,
290                },
291                Arc::clone(&param.setting_engine),
292            ));
293
294            if let Some(handler) = &*param.on_data_channel_handler.load() {
295                let mut f = handler.lock().await;
296                f(Arc::clone(&rtc_dc)).await;
297
298                param.data_channels_accepted.fetch_add(1, Ordering::SeqCst);
299
300                let mut dcs = param.data_channels.lock().await;
301                dcs.push(Arc::clone(&rtc_dc));
302            }
303
304            rtc_dc.handle_open(Arc::new(dc)).await;
305
306            if let Some(handler) = &*param.on_data_channel_opened_handler.load() {
307                let mut f = handler.lock().await;
308                f(rtc_dc).await;
309                param.data_channels_opened.fetch_add(1, Ordering::SeqCst);
310            }
311        }
312    }

334    fn calc_message_size(remote_max_message_size: usize, can_send_size: usize) -> usize {
335        if remote_max_message_size == 0 && can_send_size == 0 {
336            usize::MAX
337        } else if remote_max_message_size == 0 {
338            can_send_size
339        } else if can_send_size == 0 || can_send_size > remote_max_message_size {
340            remote_max_message_size
341        } else {
342            can_send_size
343        }
344    }

360    pub(crate) async fn collect_stats(
361        &self,
362        collector: &StatsCollector,
363        peer_connection_id: String,
364    ) {
365        let dtls_transport = self.transport();
366
367        // TODO: should this be collected?
368        dtls_transport.collect_stats(collector).await;
369
370        // data channels
371        let mut data_channels_closed = 0;
372        let data_channels = self.data_channels.lock().await;
373        for data_channel in &*data_channels {
374            match data_channel.ready_state() {
375                RTCDataChannelState::Connecting => (),
376                RTCDataChannelState::Open => (),
377                _ => data_channels_closed += 1,
378            }
379            data_channel.collect_stats(collector).await;
380        }
381
382        let mut reports = HashMap::new();
383        let peer_connection_stats =
384            PeerConnectionStats::new(self, peer_connection_id.clone(), data_channels_closed);
385        reports.insert(peer_connection_id, PeerConnection(peer_connection_stats));
386
387        // conn
388        if let Some(agent) = dtls_transport.ice_transport.gatherer.get_agent().await {
389            let stats = ICETransportStats::new("sctp_transport".to_owned(), agent);
390            reports.insert(stats.id.clone(), SCTPTransport(stats));
391        }
392
393        collector.merge(reports);
394    }

396    pub(crate) async fn generate_and_set_data_channel_id(
397        &self,
398        dtls_role: DTLSRole,
399    ) -> Result<u16> {
400        let mut id = 0u16;
401        if dtls_role != DTLSRole::Client {
402            id += 1;
403        }
404
405        // Create map of ids so we can compare without double-looping each time.
406        let mut ids_map = HashSet::new();
407        {
408            let data_channels = self.data_channels.lock().await;
409            for dc in &*data_channels {
410                ids_map.insert(dc.id());
411            }
412        }
413
414        let max = self.max_channels();
415        while id < max - 1 {
416            if ids_map.contains(&id) {
417                id += 2;
418            } else {
419                return Ok(id);
420            }
421        }
422
423        Err(Error::ErrMaxDataChannelID)
424    }
