3/// SCTPTransportState indicates the state of the SCTP transport.
4///
5/// ## Specifications
6///
7/// * [W3C]
8///
9/// [W3C]: https://w3c.github.io/webrtc-pc/#rtcsctptransportstate
10#[derive(Default, Debug, Copy, Clone, PartialEq, Eq)]
11#[repr(u8)]
12pub enum RTCSctpTransportState {
13    #[default]
14    Unspecified,
15
16    /// SCTPTransportStateConnecting indicates the SCTPTransport is in the
17    /// process of negotiating an association. This is the initial state of the
18    /// SCTPTransportState when an SCTPTransport is created.
19    Connecting,
20
21    /// SCTPTransportStateConnected indicates the negotiation of an
22    /// association is completed.
23    Connected,
24
25    /// SCTPTransportStateClosed indicates a SHUTDOWN or ABORT chunk is
26    /// received or when the SCTP association has been closed intentionally,
27    /// such as by closing the peer connection or applying a remote description
28    /// that rejects data or changes the SCTP port.
29    Closed,
30}
31
32const SCTP_TRANSPORT_STATE_CONNECTING_STR: &str = "connecting";
33const SCTP_TRANSPORT_STATE_CONNECTED_STR: &str = "connected";
34const SCTP_TRANSPORT_STATE_CLOSED_STR: &str = "closed";
35
36impl From<&str> for RTCSctpTransportState {
37    fn from(raw: &str) -> Self {
38        match raw {
39            SCTP_TRANSPORT_STATE_CONNECTING_STR => RTCSctpTransportState::Connecting,
40            SCTP_TRANSPORT_STATE_CONNECTED_STR => RTCSctpTransportState::Connected,
41            SCTP_TRANSPORT_STATE_CLOSED_STR => RTCSctpTransportState::Closed,
42            _ => RTCSctpTransportState::Unspecified,
43        }
44    }
45}
46
47impl From<u8> for RTCSctpTransportState {
48    fn from(v: u8) -> Self {
49        match v {
50            1 => RTCSctpTransportState::Connecting,
51            2 => RTCSctpTransportState::Connected,
52            3 => RTCSctpTransportState::Closed,
53            _ => RTCSctpTransportState::Unspecified,
54        }
55    }
56}
57
58impl fmt::Display for RTCSctpTransportState {
59    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
60        let s = match *self {
61            RTCSctpTransportState::Connecting => SCTP_TRANSPORT_STATE_CONNECTING_STR,
62            RTCSctpTransportState::Connected => SCTP_TRANSPORT_STATE_CONNECTED_STR,
63            RTCSctpTransportState::Closed => SCTP_TRANSPORT_STATE_CLOSED_STR,
64            RTCSctpTransportState::Unspecified => crate::UNSPECIFIED_STR,
65        };
66        write!(f, "{s}")
67    }
68}
