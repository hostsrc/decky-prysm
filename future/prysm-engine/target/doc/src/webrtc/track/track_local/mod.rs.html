23/// TrackLocalWriter is the Writer for outbound RTP Packets
24#[async_trait]
25pub trait TrackLocalWriter: fmt::Debug {
26    /// write_rtp_with_attributes encrypts a RTP packet and writes to the connection.
27    /// attributes are delivered to the interceptor chain
28    async fn write_rtp_with_attributes(
29        &self,
30        pkt: &rtp::packet::Packet,
31        attr: &Attributes,
32    ) -> Result<usize>;
33
34    /// write_rtp encrypts a RTP packet and writes to the connection
35    async fn write_rtp(&self, pkt: &rtp::packet::Packet) -> Result<usize> {
36        let attr = Attributes::new();
37        self.write_rtp_with_attributes(pkt, &attr).await
38    }
39
40    /// write encrypts and writes a full RTP packet
41    async fn write(&self, mut b: &[u8]) -> Result<usize> {
42        let pkt = rtp::packet::Packet::unmarshal(&mut b)?;
43        let attr = Attributes::new();
44        self.write_rtp_with_attributes(&pkt, &attr).await
45    }
46}

48/// TrackLocalContext is the Context passed when a TrackLocal has been Binded/Unbinded from a PeerConnection, and used
49/// in Interceptors.
50#[derive(Default, Debug, Clone)]
51pub struct TrackLocalContext {
52    pub(crate) id: String,
53    pub(crate) params: RTCRtpParameters,
54    pub(crate) ssrc: SSRC,
55    pub(crate) write_stream: Option<Arc<dyn TrackLocalWriter + Send + Sync>>,
56    pub(crate) paused: Arc<AtomicBool>,
57    pub(crate) mid: Option<SmolStr>,
58}
59
60impl TrackLocalContext {
61    /// codec_parameters returns the negotiated RTPCodecParameters. These are the codecs supported by both
62    /// PeerConnections and the SSRC/PayloadTypes
63    pub fn codec_parameters(&self) -> &[RTCRtpCodecParameters] {
64        &self.params.codecs
65    }
66
67    /// header_extensions returns the negotiated RTPHeaderExtensionParameters. These are the header extensions supported by
68    /// both PeerConnections and the SSRC/PayloadTypes
69    pub fn header_extensions(&self) -> &[RTCRtpHeaderExtensionParameters] {
70        &self.params.header_extensions
71    }
72
73    /// ssrc requires the negotiated SSRC of this track
74    /// This track may have multiple if RTX is enabled
75    pub fn ssrc(&self) -> SSRC {
76        self.ssrc
77    }
78
79    /// write_stream returns the write_stream for this TrackLocal. The implementer writes the outbound
80    /// media packets to it
81    pub fn write_stream(&self) -> Option<Arc<dyn TrackLocalWriter + Send + Sync>> {
82        self.write_stream.clone()
83    }
84
85    /// id is a unique identifier that is used for both bind/unbind
86    pub fn id(&self) -> String {
87        self.id.clone()
88    }
89
90    /// mid returns the id of media associated with the RTP stream
91    pub fn mid(&self) -> Option<SmolStr> {
92        self.mid.clone()
93    }
94
95    /// paused returns a boolean indicating whether the track is currently paused
96    pub fn paused(&self) -> Arc<AtomicBool> {
97        self.paused.clone()
98    }
99}
100/// TrackLocal is an interface that controls how the user can send media
101/// The user can provide their own TrackLocal implementations, or use
102/// the implementations in pkg/media
103#[async_trait]
104pub trait TrackLocal {
105    /// bind should implement the way how the media data flows from the Track to the PeerConnection
106    /// This will be called internally after signaling is complete and the list of available
107    /// codecs has been determined
108    async fn bind(&self, t: &TrackLocalContext) -> Result<RTCRtpCodecParameters>;
109
110    /// unbind should implement the teardown logic when the track is no longer needed. This happens
111    /// because a track has been stopped.
112    async fn unbind(&self, t: &TrackLocalContext) -> Result<()>;
113
114    /// id is the unique identifier for this Track. This should be unique for the
115    /// stream, but doesn't have to globally unique. A common example would be 'audio' or 'video'
116    /// and stream_id would be 'desktop' or 'webcam'
117    fn id(&self) -> &str;
118
119    /// RID is the RTP Stream ID for this track.
120    fn rid(&self) -> Option<&str>;
121
122    /// stream_id is the group this track belongs too. This must be unique
123    fn stream_id(&self) -> &str;
124
125    /// kind controls if this TrackLocal is audio or video
126    fn kind(&self) -> RTPCodecType;
127
128    fn as_any(&self) -> &dyn Any;
129}

175#[async_trait]
176impl TrackLocalWriter for InterceptorToTrackLocalWriter {
177    async fn write_rtp_with_attributes(
178        &self,
179        pkt: &rtp::packet::Packet,
180        attr: &Attributes,
181    ) -> Result<usize> {
182        if self.is_sender_paused() {
183            return Ok(0);
184        }
185
186        let interceptor_rtp_writer = self.interceptor_rtp_writer.lock().await;
187        if let Some(writer) = &*interceptor_rtp_writer {
188            Ok(writer.write(pkt, attr).await?)
189        } else {
190            Ok(0)
191        }
192    }
