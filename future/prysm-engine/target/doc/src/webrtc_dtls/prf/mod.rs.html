53// The premaster secret is formed as follows: if the PSK is N octets
54// long, concatenate a uint16 with the value N, N zero octets, a second
55// uint16 with the value N, and the PSK itself.
56//
57// https://tools.ietf.org/html/rfc4279#section-2
58pub(crate) fn prf_psk_pre_master_secret(psk: &[u8]) -> Vec<u8> {
59    let psk_len = psk.len();
60
61    let mut out = vec![0u8; 2 + psk_len + 2];
62
63    out.extend_from_slice(psk);
64    let be = (psk_len as u16).to_be_bytes();
65    out[..2].copy_from_slice(&be);
66    out[2 + psk_len..2 + psk_len + 2].copy_from_slice(&be);
67
68    out
69}

71pub(crate) fn prf_pre_master_secret(
72    public_key: &[u8],
73    private_key: &NamedCurvePrivateKey,
74    curve: NamedCurve,
75) -> Result<Vec<u8>> {
76    match curve {
77        NamedCurve::P256 => elliptic_curve_pre_master_secret(public_key, private_key, curve),
78        NamedCurve::P384 => elliptic_curve_pre_master_secret(public_key, private_key, curve),
79        NamedCurve::X25519 => elliptic_curve_pre_master_secret(public_key, private_key, curve),
80        _ => Err(Error::ErrInvalidNamedCurve),
81    }
82}

84fn elliptic_curve_pre_master_secret(
85    public_key: &[u8],
86    private_key: &NamedCurvePrivateKey,
87    curve: NamedCurve,
88) -> Result<Vec<u8>> {
89    match curve {
90        NamedCurve::P256 => {
91            let pub_key = p256::EncodedPoint::from_bytes(public_key)?;
92            let public = p256::PublicKey::from_sec1_bytes(pub_key.as_ref())?;
93            if let NamedCurvePrivateKey::EphemeralSecretP256(secret) = private_key {
94                return Ok(secret.diffie_hellman(&public).raw_secret_bytes().to_vec());
95            }
96        }
97        NamedCurve::P384 => {
98            let pub_key = p384::EncodedPoint::from_bytes(public_key)?;
99            let public = p384::PublicKey::from_sec1_bytes(pub_key.as_ref())?;
100            if let NamedCurvePrivateKey::EphemeralSecretP384(secret) = private_key {
101                return Ok(secret.diffie_hellman(&public).raw_secret_bytes().to_vec());
102            }
103        }
104        NamedCurve::X25519 => {
105            if public_key.len() != 32 {
106                return Err(Error::Other("Public key is not 32 len".into()));
107            }
108            let pub_key: [u8; 32] = public_key.try_into().unwrap();
109            let public = x25519_dalek::PublicKey::from(pub_key);
110            if let NamedCurvePrivateKey::StaticSecretX25519(secret) = private_key {
111                return Ok(secret.diffie_hellman(&public).as_bytes().to_vec());
112            }
113        }
114        _ => return Err(Error::ErrInvalidNamedCurve),
115    }
116    Err(Error::ErrNamedCurveAndPrivateKeyMismatch)
117}

119//  This PRF with the SHA-256 hash function is used for all cipher suites
120//  defined in this document and in TLS documents published prior to this
121//  document when TLS 1.2 is negotiated.  New cipher suites MUST explicitly
122//  specify a PRF and, in general, SHOULD use the TLS PRF with SHA-256 or a
123//  stronger standard hash function.
124//
125//     P_hash(secret, seed) = HMAC_hash(secret, A(1) + seed) +
126//                            HMAC_hash(secret, A(2) + seed) +
127//                            HMAC_hash(secret, A(3) + seed) + ...
128//
129//  A() is defined as:
130//
131//     A(0) = seed
132//     A(i) = HMAC_hash(secret, A(i-1))
133//
134//  P_hash can be iterated as many times as necessary to produce the
135//  required quantity of data.  For example, if P_SHA256 is being used to
136//  create 80 bytes of data, it will have to be iterated three times
137//  (through A(3)), creating 96 bytes of output data; the last 16 bytes
138//  of the final iteration will then be discarded, leaving 80 bytes of
139//  output data.
140//
141// https://tools.ietf.org/html/rfc4346w
142fn hmac_sha(h: CipherSuiteHash, key: &[u8], data: &[u8]) -> Result<Vec<u8>> {
143    let mut mac = match h {
144        CipherSuiteHash::Sha256 => {
145            HmacSha256::new_from_slice(key).map_err(|e| Error::Other(e.to_string()))?
146        }
147    };
148    mac.update(data);
149    let result = mac.finalize();
150    let code_bytes = result.into_bytes();
151    Ok(code_bytes.to_vec())
152}

154pub(crate) fn prf_p_hash(
155    secret: &[u8],
156    seed: &[u8],
157    requested_length: usize,
158    h: CipherSuiteHash,
159) -> Result<Vec<u8>> {
160    let mut last_round = seed.to_vec();
161    let mut out = vec![];
162
163    let iterations = ((requested_length as f64) / (h.size() as f64)).ceil() as usize;
164    for _ in 0..iterations {
165        last_round = hmac_sha(h, secret, &last_round)?;
166
167        let mut last_round_seed = last_round.clone();
168        last_round_seed.extend_from_slice(seed);
169        let with_secret = hmac_sha(h, secret, &last_round_seed)?;
170
171        out.extend_from_slice(&with_secret);
172    }
173
174    Ok(out[..requested_length].to_vec())
175}

199pub(crate) fn prf_encryption_keys(
200    master_secret: &[u8],
201    client_random: &[u8],
202    server_random: &[u8],
203    prf_mac_len: usize,
204    prf_key_len: usize,
205    prf_iv_len: usize,
206    h: CipherSuiteHash,
207) -> Result<EncryptionKeys> {
208    let mut seed = PRF_KEY_EXPANSION_LABEL.as_bytes().to_vec();
209    seed.extend_from_slice(server_random);
210    seed.extend_from_slice(client_random);
211
212    let material = prf_p_hash(
213        master_secret,
214        &seed,
215        (2 * prf_mac_len) + (2 * prf_key_len) + (2 * prf_iv_len),
216        h,
217    )?;
218    let mut key_material = &material[..];
219
220    let client_mac_key = key_material[..prf_mac_len].to_vec();
221    key_material = &key_material[prf_mac_len..];
222
223    let server_mac_key = key_material[..prf_mac_len].to_vec();
224    key_material = &key_material[prf_mac_len..];
225
226    let client_write_key = key_material[..prf_key_len].to_vec();
227    key_material = &key_material[prf_key_len..];
228
229    let server_write_key = key_material[..prf_key_len].to_vec();
230    key_material = &key_material[prf_key_len..];
231
232    let client_write_iv = key_material[..prf_iv_len].to_vec();
233    key_material = &key_material[prf_iv_len..];
234
235    let server_write_iv = key_material[..prf_iv_len].to_vec();
236
237    Ok(EncryptionKeys {
238        master_secret: master_secret.to_vec(),
239        client_mac_key,
240        server_mac_key,
241        client_write_key,
242        server_write_key,
243        client_write_iv,
244        server_write_iv,
245    })
246}

248pub(crate) fn prf_verify_data(
249    master_secret: &[u8],
250    handshake_bodies: &[u8],
251    label: &str,
252    h: CipherSuiteHash,
253) -> Result<Vec<u8>> {
254    let mut hasher = match h {
255        CipherSuiteHash::Sha256 => Sha256::new(),
256    };
257    hasher.update(handshake_bodies);
258    let result = hasher.finalize();
259    let mut seed = label.as_bytes().to_vec();
260    seed.extend_from_slice(&result);
261
262    prf_p_hash(master_secret, &seed, 12, h)
263}

291// compute the MAC using HMAC-SHA1
292pub(crate) fn prf_mac(
293    epoch: u16,
294    sequence_number: u64,
295    content_type: ContentType,
296    protocol_version: ProtocolVersion,
297    payload: &[u8],
298    key: &[u8],
299) -> Result<Vec<u8>> {
300    let mut hmac = HmacSha1::new_from_slice(key).map_err(|e| Error::Other(e.to_string()))?;
301
302    let mut msg = vec![0u8; 13];
303    msg[..2].copy_from_slice(&epoch.to_be_bytes());
304    msg[2..8].copy_from_slice(&sequence_number.to_be_bytes()[2..]);
305    msg[8] = content_type as u8;
306    msg[9] = protocol_version.major;
307    msg[10] = protocol_version.minor;
308    msg[11..].copy_from_slice(&(payload.len() as u16).to_be_bytes());
309
310    hmac.update(&msg);
311    hmac.update(payload);
312    let result = hmac.finalize();
313
314    Ok(result.into_bytes().to_vec())
