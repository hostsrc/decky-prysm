38            master_key,
39            master_salt,
40            0,
41            master_salt.len(),
42        )?;
43        let srtcp_session_salt = aes_cm_key_derivation(
44            LABEL_SRTCP_SALT,
45            master_key,
46            master_salt,
47            0,

49        )?;

52
53        let srtp_session_auth_tag = aes_cm_key_derivation(
54            LABEL_SRTP_AUTHENTICATION_TAG,
55            master_key,
56            master_salt,
57            0,
58            auth_key_len,
59        )?;
60        let srtcp_session_auth_tag = aes_cm_key_derivation(

62            master_key,

69            .map_err(|e| Error::Other(e.to_string()))?;

73        Ok(Self {

75            srtp_session_salt,
76            srtp_session_auth,
77            srtcp_session_salt,

82    /// https://tools.ietf.org/html/rfc3711#section-4.2
83    /// In the case of SRTP, M SHALL consist of the Authenticated
84    /// Portion of the packet (as specified in Figure 1) concatenated with
85    /// the roc, M = Authenticated Portion || roc;
86    ///
87    /// The pre-defined authentication transform for SRTP is HMAC-SHA1
88    /// [RFC2104].  With HMAC-SHA1, the SRTP_PREFIX_LENGTH (Figure 3) SHALL
89    /// be 0.  For SRTP (respectively SRTCP), the HMAC SHALL be applied to
90    /// the session authentication key and M as specified above, i.e.,
91    /// HMAC(k_a, M).  The HMAC output SHALL then be truncated to the n_tag
92    /// left-most bits.
93    /// - Authenticated portion of the packet is everything BEFORE MKI
94    /// - k_a is the session message authentication key
95    /// - n_tag is the bit-length of the output authentication tag
96    fn generate_srtp_auth_tag(&self, buf: &[u8], roc: u32) -> [u8; 20] {
97        let mut signer = self.srtp_session_auth.clone();
98
99        signer.update(buf);
100
101        // For SRTP only, we need to hash the rollover counter as well.
102        signer.update(&roc.to_be_bytes());
103
104        signer.finalize().into_bytes().into()
105    }

120
121        signer.update(buf);
122

124    }

126    fn get_rtcp_index(&self, input: &[u8]) -> usize {
