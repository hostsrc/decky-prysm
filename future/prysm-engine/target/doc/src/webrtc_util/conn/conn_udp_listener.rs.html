22/// listener is used in the [DTLS](https://github.com/webrtc-rs/dtls) and
23/// [SCTP](https://github.com/webrtc-rs/sctp) transport to provide a connection-oriented
24/// listener over a UDP.
25struct ListenerImpl {
26    pconn: Arc<dyn Conn + Send + Sync>,
27    accepting: Arc<AtomicBool>,
28    accept_ch_tx: Arc<Mutex<Option<mpsc::Sender<Arc<UdpConn>>>>>,
29    done_ch_tx: Arc<Mutex<Option<watch::Sender<()>>>>,
30    ch_rx: Arc<Mutex<AcceptDoneCh>>,
31    conns: Arc<Mutex<HashMap<String, Arc<UdpConn>>>>,
32}
33
34#[async_trait]
35impl Listener for ListenerImpl {
36    /// accept waits for and returns the next connection to the listener.
37    async fn accept(&self) -> Result<(Arc<dyn Conn + Send + Sync>, SocketAddr)> {
38        let (accept_ch_rx, done_ch_rx) = &mut *self.ch_rx.lock().await;
39
40        tokio::select! {
41            c = accept_ch_rx.recv() =>{
42                if let Some(c) = c{
43                    let raddr = c.raddr;
44                    Ok((c, raddr))
45                }else{
46                    Err(Error::ErrClosedListenerAcceptCh)
47                }
48            }
49            _ = done_ch_rx.changed() =>  Err(Error::ErrClosedListener),
50        }
51    }
52
53    /// close closes the listener.
54    /// Any blocked Accept operations will be unblocked and return errors.
55    async fn close(&self) -> Result<()> {
56        if self.accepting.load(Ordering::SeqCst) {
57            self.accepting.store(false, Ordering::SeqCst);
58            {
59                let mut done_ch = self.done_ch_tx.lock().await;
60                done_ch.take();
61            }
62            {
63                let mut accept_ch = self.accept_ch_tx.lock().await;
64                accept_ch.take();
65            }
66        }
67
68        Ok(())
69    }
70
71    /// Addr returns the listener's network address.
72    async fn addr(&self) -> Result<SocketAddr> {
73        self.pconn.local_addr()
74    }
75}

77/// ListenConfig stores options for listening to an address.
78#[derive(Default)]
79pub struct ListenConfig {
80    /// Backlog defines the maximum length of the queue of pending
81    /// connections. It is equivalent of the backlog argument of
82    /// POSIX listen function.
83    /// If a connection request arrives when the queue is full,
84    /// the request will be silently discarded, unlike TCP.
85    /// Set zero to use default value 128 which is same as Linux default.
86    pub backlog: usize,
87
88    /// AcceptFilter determines whether the new conn should be made for
89    /// the incoming packet. If not set, any packet creates new conn.
90    pub accept_filter: Option<AcceptFilterFn>,
91}

93pub async fn listen<A: ToSocketAddrs>(laddr: A) -> Result<impl Listener> {
94    ListenConfig::default().listen(laddr).await
95}

97impl ListenConfig {
98    /// Listen creates a new listener based on the ListenConfig.
99    pub async fn listen<A: ToSocketAddrs>(&mut self, laddr: A) -> Result<impl Listener> {
100        if self.backlog == 0 {
101            self.backlog = DEFAULT_LISTEN_BACKLOG;
102        }
103
104        let pconn = Arc::new(UdpSocket::bind(laddr).await?);
105        let (accept_ch_tx, accept_ch_rx) = mpsc::channel(self.backlog);
106        let (done_ch_tx, done_ch_rx) = watch::channel(());
107
108        let l = ListenerImpl {
109            pconn,
110            accepting: Arc::new(AtomicBool::new(true)),
111            accept_ch_tx: Arc::new(Mutex::new(Some(accept_ch_tx))),
112            done_ch_tx: Arc::new(Mutex::new(Some(done_ch_tx))),
113            ch_rx: Arc::new(Mutex::new((accept_ch_rx, done_ch_rx.clone()))),
114            conns: Arc::new(Mutex::new(HashMap::new())),
115        };
116
117        let pconn = Arc::clone(&l.pconn);
118        let accepting = Arc::clone(&l.accepting);
119        let accept_filter = self.accept_filter.take();
120        let accept_ch_tx = Arc::clone(&l.accept_ch_tx);
121        let conns = Arc::clone(&l.conns);
122        tokio::spawn(async move {
123            ListenConfig::read_loop(
124                done_ch_rx,
125                pconn,
126                accepting,
127                accept_filter,
128                accept_ch_tx,
129                conns,
130            )
131            .await;
132        });
133
134        Ok(l)
135    }
136
137    /// read_loop has to tasks:
138    /// 1. Dispatching incoming packets to the correct Conn.
139    ///    It can therefore not be ended until all Conns are closed.
140    /// 2. Creating a new Conn when receiving from a new remote.
141    async fn read_loop(
142        mut done_ch_rx: watch::Receiver<()>,
143        pconn: Arc<dyn Conn + Send + Sync>,
144        accepting: Arc<AtomicBool>,
145        accept_filter: Option<AcceptFilterFn>,
146        accept_ch_tx: Arc<Mutex<Option<mpsc::Sender<Arc<UdpConn>>>>>,
147        conns: Arc<Mutex<HashMap<String, Arc<UdpConn>>>>,
148    ) {
149        let mut buf = vec![0u8; RECEIVE_MTU];
150
151        loop {
152            tokio::select! {
153                _ = done_ch_rx.changed() => {
154                    break;
155                }
156                result = pconn.recv_from(&mut buf) => {
157                    match result {
158                        Ok((n, raddr)) => {
159                            let udp_conn = match ListenConfig::get_udp_conn(
160                                &pconn,
161                                &accepting,
162                                &accept_filter,
163                                &accept_ch_tx,
164                                &conns,
165                                raddr,
166                                &buf[..n],
167                            )
168                            .await
169                            {
170                                Ok(conn) => conn,
171                                Err(_) => continue,
172                            };
173
174                            if let Some(conn) = udp_conn {
175                                let _ = conn.buffer.write(&buf[..n]).await;
176                            }
177                        }
178                        Err(err) => {
179                            log::warn!("ListenConfig pconn.recv_from error: {}", err);
180                            break;
181                        }
182                    };
183                }
184            }
185        }
186    }
187
188    async fn get_udp_conn(
189        pconn: &Arc<dyn Conn + Send + Sync>,
190        accepting: &Arc<AtomicBool>,
191        accept_filter: &Option<AcceptFilterFn>,
192        accept_ch_tx: &Arc<Mutex<Option<mpsc::Sender<Arc<UdpConn>>>>>,
193        conns: &Arc<Mutex<HashMap<String, Arc<UdpConn>>>>,
194        raddr: SocketAddr,
195        buf: &[u8],
196    ) -> Result<Option<Arc<UdpConn>>> {
197        {
198            let m = conns.lock().await;
199            if let Some(conn) = m.get(raddr.to_string().as_str()) {
200                return Ok(Some(conn.clone()));
201            }
202        }
203
204        if !accepting.load(Ordering::SeqCst) {
205            return Err(Error::ErrClosedListener);
206        }
207
208        if let Some(f) = accept_filter {
209            if !(f(buf).await) {
210                return Ok(None);
211            }
212        }
213
214        let udp_conn = Arc::new(UdpConn::new(Arc::clone(pconn), raddr));
215        {
216            let accept_ch = accept_ch_tx.lock().await;
217            if let Some(tx) = &*accept_ch {
218                if tx.try_send(Arc::clone(&udp_conn)).is_err() {
219                    return Err(Error::ErrListenQueueExceeded);
220                }
221            } else {
222                return Err(Error::ErrClosedListenerAcceptCh);
223            }
224        }
225
226        {
227            let mut m = conns.lock().await;
228            m.insert(raddr.to_string(), Arc::clone(&udp_conn));
229        }
230
231        Ok(Some(udp_conn))
232    }
233}

235/// UdpConn augments a connection-oriented connection over a UdpSocket
236pub struct UdpConn {
237    pconn: Arc<dyn Conn + Send + Sync>,
238    raddr: SocketAddr,
239    buffer: Buffer,
240}
241
242impl UdpConn {
243    fn new(pconn: Arc<dyn Conn + Send + Sync>, raddr: SocketAddr) -> Self {
244        UdpConn {
245            pconn,
246            raddr,
247            buffer: Buffer::new(0, 0),
248        }
249    }
250}
251
252#[async_trait]
253impl Conn for UdpConn {
254    async fn connect(&self, addr: SocketAddr) -> Result<()> {
255        self.pconn.connect(addr).await
256    }
257
258    async fn recv(&self, buf: &mut [u8]) -> Result<usize> {
259        Ok(self.buffer.read(buf, None).await?)
260    }
261
262    async fn recv_from(&self, buf: &mut [u8]) -> Result<(usize, SocketAddr)> {
263        let n = self.buffer.read(buf, None).await?;
264        Ok((n, self.raddr))
265    }
266
267    async fn send(&self, buf: &[u8]) -> Result<usize> {
268        self.pconn.send_to(buf, self.raddr).await
269    }
270
271    async fn send_to(&self, buf: &[u8], target: SocketAddr) -> Result<usize> {
272        self.pconn.send_to(buf, target).await
273    }
274
275    fn local_addr(&self) -> Result<SocketAddr> {
276        self.pconn.local_addr()
277    }
278
279    fn remote_addr(&self) -> Option<SocketAddr> {
280        Some(self.raddr)
281    }
282
283    async fn close(&self) -> Result<()> {
284        Ok(())
285    }
286
287    fn as_any(&self) -> &(dyn std::any::Any + Send + Sync) {
288        self
289    }
