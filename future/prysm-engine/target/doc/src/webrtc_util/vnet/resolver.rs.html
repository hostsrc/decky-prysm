15#[derive(Default)]
16pub(crate) struct Resolver {
17    parent: Option<Weak<Mutex<Resolver>>>,
18    hosts: HashMap<String, IpAddr>,
19}
20
21impl Resolver {
22    pub(crate) fn new() -> Self {
23        let mut r = Resolver {
24            parent: None,
25            hosts: HashMap::new(),
26        };
27
28        if let Err(err) = r.add_host("localhost".to_owned(), "127.0.0.1".to_owned()) {
29            log::warn!("failed to add localhost to Resolver: {}", err);
30        }
31        r
32    }
33
34    pub(crate) fn set_parent(&mut self, p: Weak<Mutex<Resolver>>) {
35        self.parent = Some(p);
36    }
37
38    pub(crate) fn add_host(&mut self, name: String, ip_addr: String) -> Result<()> {
39        if name.is_empty() {
40            return Err(Error::ErrHostnameEmpty);
41        }
42        let ip = IpAddr::from_str(&ip_addr)?;
43        self.hosts.insert(name, ip);
44
45        Ok(())
46    }
47
48    pub(crate) fn lookup(
49        &self,
50        host_name: String,
51    ) -> Pin<Box<dyn Future<Output = Option<IpAddr>> + Send + 'static>> {
52        if let Some(ip) = self.hosts.get(&host_name) {
53            let ip2 = *ip;
54            return Box::pin(async move { Some(ip2) });
55        }
56
57        // mutex must be unlocked before calling into parent Resolver
58        if let Some(parent) = self.parent.clone().and_then(|p| p.upgrade()).clone() {
59            Box::pin(async move {
60                let p = parent.lock().await;
61                p.lookup(host_name).await
62            })
63        } else {
64            Box::pin(async move { None })
65        }
66    }
67}
